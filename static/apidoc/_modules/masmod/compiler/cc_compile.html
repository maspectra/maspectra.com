    def __init__(
        self,
        logger: ILogger,
        cc_source_code: str,
        src_include_dir: pathlib.Path,
        eigen_include_dir: pathlib.Path,
        temp_dir: pathlib.Path | None = None,
        uid: str | None = None
    ) -> None:
        self._logger = logger
        if temp_dir is None:
            temp_dir = pathlib.Path(tempfile.gettempdir()).joinpath("mas")
        if not temp_dir.exists():
            temp_dir.mkdir()
        self._temp_dir = temp_dir
        self._source_code = cc_source_code

        self.platform_system = platform.system()
        # TODO: detect g++
        # _toolchain = "gcc"
        _cxx_compiler = shutil.which("g++")
        cxx_compiler_env_var = os.getenv("MASMOD_CXX_COMPILER")
        if cxx_compiler_env_var:
            cxx_compiler_path = pathlib.Path(cxx_compiler_env_var)
            if not cxx_compiler_path.exists():
                logger.error(f"⚠️ MASMOD_CXX_COMPILER {cxx_compiler_env_var} not found, fallback to global g++\n")
            else:
                _cxx_compiler = cxx_compiler_path.as_posix()

        if _cxx_compiler is None:
            logger.error("⚠️ g++ not found, please install g++ first or set MASMOD_CXX_COMPILER=/path/to/g++\n")
            raise RuntimeError("g++ not found")

        self.cxx_compiler = _cxx_compiler

        self._SRC_ROOT = src_include_dir
        self._EIGEN_ROOT = eigen_include_dir

        if uid is None:
            uid = str(sha1.SHA1.with_content(cc_source_code))
        temp_file_cc = self._temp_dir.joinpath(f"{uid}.cc")
        with open(temp_file_cc, mode="w", encoding="utf-8") as f:
            f.write(f"// Auto Generate at {datetime.now()}\n")
            f.write(cc_source_code)

        self._uid = uid
        self.temp_file_cc = temp_file_cc
        self.temp_file_obj = self._temp_dir.joinpath(f"{self._uid}.o")
        suffix: str
        if self.platform_system == "Windows":
            suffix = "dll"
        elif self.platform_system == "Darwin":
            suffix = "so"
        else:
            raise NotImplementedError
        self.temp_file_dylib = self._temp_dir.joinpath(f"{self._uid}.{suffix}")


[docs]    def compile(self) -> CompileResult:
        if self.temp_file_obj.exists() and self.temp_file_dylib.exists():
            self._logger.info("➡️ Since build cache already exists, skip compile\n")
        else:
            self._do_build_obj()
            self._do_build_dylib()
            self._logger.info("✅ Compilation Finished\n")

        return CompileResult(target_file=self.temp_file_dylib)


    def _do_build_obj(self) -> None:
        self._logger.info(f"🔧 CXX compiler {self.cxx_compiler}\n")
        self._logger.info("📦 Compiling build target...\n")

        flags = [
            "-std=c++17",
            "-O3",
            f"-I\"{self._SRC_ROOT.as_posix()}\"",
            f"-I\"{self._EIGEN_ROOT.as_posix()}\"",
            f"-c \"{self.temp_file_cc.as_posix()}\"",
            f"-o \"{self.temp_file_obj.as_posix()}\"",
        ]
        self._exec(f"\"{self.cxx_compiler}\" {' '.join(flags)}")

    def _do_build_dylib(self) -> None:
        self._logger.info("🔗 Linking dynamic library...\n")

        flags = [
            f"\"{self.temp_file_obj.as_posix()}\"",
            "-shared",
            "-fPIC",
            f"-o \"{self.temp_file_dylib.as_posix()}\"",
        ]
        self._exec(f"\"{self.cxx_compiler}\" {' '.join(flags)}")

    def _exec(self, cmd: str) -> str:
        try:
            output = subprocess.check_output(cmd, shell=True, encoding="utf-8")
        except Exception as e:
            raise RuntimeError(f"Failed to execute command\n $ {cmd}") from e

        return output
