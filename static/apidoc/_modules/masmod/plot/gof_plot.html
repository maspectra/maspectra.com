[docs]def trend_line(x: npt.NDArray[np.float_],
                y: npt.NDArray[np.float_]) -> tuple[npt.NDArray[np.float_], npt.NDArray[np.float_]]:
    if ((len(np.unique(x)) <= 1) and (len(np.unique(y)) <= 1)):
        k = 0
        b = 0
    else:
        k, b = np.polyfit(x, y, deg=1)  # 计算拟合线
    x_max = np.max(x)
    y_max = k * x_max + b
    x_min = np.min(x)
    y_min = k * x_min + b

    return np.array([x_min, x_max]), np.array([y_min, y_max])




[docs]def diagonal_range(x: npt.NDArray[np.float_],
                    y: npt.NDArray[np.float_],
                    proportion: float = 1.5) -> tuple[npt.NDArray[np.float_], npt.NDArray[np.float_]]:
    if (np.max(x) > np.max(y)):
        x_max = np.max(x) + (np.max(x) - np.mean(x)) * proportion
        y_max = x_max
    else:
        y_max = np.max(y) + (np.max(y) - np.mean(y)) * proportion
        x_max = y_max
    if (np.min(x) < np.min(y)):
        x_min = np.min(x) - (np.mean(x) - np.min(x)) * proportion
        y_min = x_min
    else:
        y_min = np.min(y) - (np.mean(y) - np.min(y)) * proportion
        x_min = y_min
    return np.array([x_min, x_max]), np.array([y_min, y_max])

            curve_x = [x_lim[0] + x * (x_lim[1] - x_lim[0]) / 500 for x in range(500)]
            curve_y = stats.gaussian_kde(wres_data)(curve_x)
            trace_wres_trend = go.Scatter(
                x=curve_x, y=curve_y, mode='lines', hoverinfo='x+y', marker=dict(color='#36556F'), showlegend=False
            )
        except:
            trace_wres_trend = go.Scatter()
        fig.add_trace(trace_wres_trend, row=3, col=2)

        fig['layout']['xaxis6'].update(
            title=self._label_wres,
            dtick=1,
            range=x_lim,
            showgrid=False,
            zeroline=False,
        )
        fig['layout']['yaxis6'].update(title="Density", showgrid=False, zeroline=True)

        height: float = 350 * 3
        width: float = 350 * 2 * 1.15
        fig['layout'].update(shapes=shapes)
        fig.update_xaxes(
            showline=True,
            linecolor='#000',
            linewidth=2,
            mirror=True,
            ticks='outside',
        )
        fig.update_yaxes(
            showline=True,
            linecolor='#000',
            linewidth=2,
            mirror=True,
            ticks='outside',
        )

        fig.update_layout(
            height=height,
            width=width,
            title={
                "text": self._title, 'y': 0.98, 'x': 0.5
            },
            title_font={"size": 20},
            plot_bgcolor='#fff'
        )

        return fig
