[docs]def polarLUV_to_LUV(H: float, L: float, C: float) -> List[float]:
    H = np.pi / 180. * H
    L = L
    U = C * np.cos(H)
    V = C * np.sin(H)
    return [L, U, V]




[docs]def LUV_to_XYZ(L: float, U: float, V: float) -> List[float]:
    eps = np.finfo(float).eps * 10
    KAPPA = 903.2962962962963
    XN, YN, ZN = 95.047, 100, 108.883
    Y = YN * (np.power((L + 16.) / 116., 3.) if L > 8. else L / KAPPA)
    L = np.fmax(eps, L)
    t = XN + YN + ZN
    x = XN / t
    y = YN / t
    uN, vN = 2.0 * x / (6. * y - x + 1.5), 4.5 * y / (6. * y - x + 1.5)
    u = U / (13. * L) + uN
    v = V / (13. * L) + vN
    X = 9.0 * Y * u / (4 * v)
    Z = -X / 3. - 5. * Y + 3. * Y / v
    return [X, Y, Z]




[docs]def XYZ_to_RGB(X: float, Y: float, Z: float) -> List[float]:
    gamma = 2.4
    YN = 100
    # xyz to rgb
    a = [
        (3.240479 * X - 1.537150 * Y - 0.498535 * Z) / YN,  # r
        (-0.969256 * X + 1.875992 * Y + 0.041556 * Z) / YN,  # g
        (0.055648 * X - 0.204043 * Y + 1.057311 * Z) / YN  # b
    ]
    # rgb->srgb
    val: float
    for i, val in enumerate(a):
        if val > 0.00304:
            val = 1.055 * np.power(val, (1. / gamma)) - 0.055
        else:
            val = 12.92 * val
        if val < 0:
            val = 0
        elif val > 1:
            val = 1
        a[i] = val
    R, G, B = round(a[0] * 255), round(a[1] * 255), round(a[2] * 255)
    return [R, G, B]




[docs]def get_color(n: int) -> List[str]:
    colors: List[str] = []
    L = 65
    C = 100
    for H in np.linspace(15, 375, n + 1):
        L, U, V = polarLUV_to_LUV(H, L, C)
        X, Y, Z = LUV_to_XYZ(L, U, V)
        R, G, B = XYZ_to_RGB(X, Y, Z)
        color = f"rgb({R},{G},{B})"
        colors.append(color)

def _single_var_plot(df: pd.DataFrame, xvar: str, yvar: str) -> go.Scatter:

def _single_box_plot(data: pd.Series, mode: Literal["x", "y"], name: str, color: str | None = None) -> go.Box:
    trace = None
    if mode == 'x':
        trace = go.Box(
            x=data,
            name=str(name),
            line_width=1,
            marker=dict(size=4, color=color),
        )
    elif mode == 'y':
        trace = go.Box(
            y=data,
            name=str(name),
            line_width=1,
            marker=dict(size=4, color=color),
        )
    return trace


def _trace_linear_regression(df: pd.DataFrame, xvar: str, yvar: str) -> go.Scatter:
    is_x_null = ~df[xvar].isnull()
    is_y_null = ~df[yvar].isnull()
    data_x = df[xvar][is_x_null & is_y_null]
    data_y = df[yvar][is_x_null & is_y_null]

    dat_x = data_x.values.reshape(-1, 1)  # type: ignore
    dat_y = data_y.values.reshape(-1, 1)  # type: ignore

    regr = LinearRegression()

    regr.fit(dat_x, dat_y)

    y: np.ndarray = regr.predict(dat_x)

    fig_trace = go.Scatter(
        x=dat_x.squeeze(), y=y.squeeze(), hoverinfo='none', mode="lines", line={
            'color': 'red', 'width': 1.5
        }
    )
    return fig_trace


def _trace_smoothing_line(df: pd.DataFrame, xvar: str, yvar: str) -> go.Scatter:
    df = df[[xvar, yvar]].dropna()

    dat_x = df[xvar].values.reshape(-1, 1)  # type: ignore
    dat_y = df[yvar].values.reshape(-1, 1)  # type: ignore

    lowess_dat = lowess(dat_y.squeeze(), dat_x.squeeze())

    fig_trace = go.Scatter(
        x=lowess_dat[:, 0],
        y=lowess_dat[:, 1],
        hoverinfo='none',
        mode="lines",
        line={
            # 'shape': 'spline',
            'color': 'red',
            'width': 1.5
        }
    )
    return fig_trace


def _scatter_by_group(data_x: pd.Series, data_y: pd.Series, id: Any, subject_name: str | None) -> go.Scatter:
    _hover_text = (subject_name + "=") if subject_name else "ID="
    trace = go.Scatter(

        y=data_y,
        mode="lines+markers",
        hoverinfo="x+y+text",
        showlegend=False,

            'symbol': 'circle-open', "color": "#4682b4"

    )
    return trace


def _add_pd_data(
    df: pd.DataFrame,
    x_dvid: str | int | float,
    y_dvid: str | int | float,
    dv_name: str,
    idv_name: str,
    dvid_name: str
) -> pd.DataFrame:
    pk_data = df[df[dvid_name] == x_dvid]
    pd_data = df[df[dvid_name] == y_dvid]
    pd_result = pd_data.copy()
    for time in pk_data[idv_name].values:
        if time not in pd_data[idv_name].values:
            result_1: pd.Series = (pd_data[idv_name] < time)
            result_2: pd.Series = (pd_data[idv_name] > time)
            if result_1.any() and result_2.any():
                start = pd_data[result_1].iloc[-1]
                end = pd_data[result_2].iloc[0]
                # 计算补点
                value = (end[dv_name] - start[dv_name]) / (end[idv_name] - start[idv_name]) * time + \
                        start[dv_name]
                # 重组补点值
                insert_value = start.copy()
                insert_value[idv_name] = time
                insert_value[dvid_name] = y_dvid
                insert_value[dv_name] = value
                pd_result = pd.concat([pd_result, insert_value.to_frame().T], axis="index", ignore_index=True)
            else:
                break

    return pd.concat([pk_data, pd_result])

[docs]class ContourPlot(AbstractGroupbyablePlotlyPlot):

    def __init__(
        self,
        data: pd.DataFrame,
        subject_id_name: str | None,
        idv_name: str,
        dv_name: str,
        title: str = "",
        x_scale: Literal["linear", "log"] = "linear",

    ) -> None:
        super().__init__(data=data, title=title, subject_id_name=subject_id_name, meta={})
        self._idv_name = idv_name
        self._dv_name = dv_name
        self._x_scale: Literal["linear", "log"] = x_scale
        self._y_scale: Literal["linear", "log"] = y_scale

    def _make_plot(self) -> go.Figure:
        cols = 1
        rows = 1
        nums = 0

        # pk pd 存在情况
        x_var_name = self._idv_name
        y_var_name = self._dv_name
        id_var_name = self._id_name
        data_df = self._data

        if id_var_name:
            if x_var_name != y_var_name:
                data_df = data_df[[id_var_name, x_var_name, y_var_name]]
            else:
                data_df = data_df[[id_var_name, x_var_name]]
            result = data_df
        else:
            if x_var_name != y_var_name:
                data_df = data_df[[x_var_name, y_var_name]]
            else:
                data_df = data_df[[x_var_name]]
            result = data_df

        # result = result.dropna()
        fig = make_subplots(rows=rows, cols=cols)

        for (id, frame_id) in result.groupby(id_var_name):
            # 重组数据
            nums += 1
            fig.add_trace(
                _scatter_by_group(frame_id[x_var_name], frame_id[y_var_name], id=id, subject_name=self._id_name),
                row=1,
                col=1
            )

        fig.update_layout(plot_bgcolor='#fff')
        # 设置轴的样式
        for axis_index in range(1, rows * cols + 1):
            fig['layout'][f'xaxis{axis_index}'].update(
                title=x_var_name,
                showline=True,
                linecolor='#000',
                linewidth=1,
                showgrid=False,
                zeroline=False,
                mirror=True,
                ticks='outside',
                type=self._x_scale
            )

            fig['layout'][f'yaxis{axis_index}'].update(
                title=y_var_name,
                showline=True,
                matches='y',
                type=self._y_scale,
                linecolor='#000',
                linewidth=1,
                showgrid=False,
                zeroline=False,
                mirror=True,
                ticks='outside'
            )

        # count_df = result.groupby("id")["time"].count()
        annotations = [
            dict(
                visible=False,
                xref="paper",
                yref="paper",
                x=0.98,
                y=0.98,
                text=
                "N Subject: {num} <br>Average dosing times: {amt} <br>Observation record Total/Average/Min/Max:{total}/{avg}/{min}/{max}"
                .format(
                    num=nums,
                    amt=1,
                    total=result.shape[0],
                    c=result.shape[0],
                    avg=result.groupby(id_var_name)[x_var_name].count().sum() / result.shape[0],
                    min=result.groupby(id_var_name)[x_var_name].count().min(),
                    max=result.groupby(id_var_name)[x_var_name].count().max(),
                ),
                showarrow=False,
                align="left",
                bgcolor="#f9f9f9",
                bordercolor="grey"
            )
        ]
        fig.update_layout(
            height=600, width=600, annotations=annotations, title={
                "text": self._title, 'y': 0.92, 'x': 0.5
            }
        )

        return fig

        )

                    col=1
                )
