from typing import Literal

import numpy as np
import numpy.typing as npt
import pandas as pd
import plotly.graph_objs as go

from mas.libs.masmod.plot.abstract_groupby import AbstractGroupbyablePlotlyPlot


def _individual_curve(x: pd.Series | np.ndarray, y: pd.Series | np.ndarray, name: str = "") -> go.Scatter:

    return go.Scatter(
        x=x,
        y=y,
        name=f'line:{name}',
        mode='lines',
        showlegend=False,
        line=dict(color='rgba(55,122,222,0.7)', width=1.5),
        text=f'{name}',
        hoverinfo='text',
    )



[docs]class SimulateIndividualPlot(AbstractGroupbyablePlotlyPlot):

    def __init__(
        self,
        data: pd.DataFrame,
        subject_id_name: str | None,
        idv_name: str,
        dv_name: str,
        title: str = "",
        x_scale: Literal["linear", "log"] = "linear",
        y_scale: Literal["linear", "log"] = "linear",
    ) -> None:
        super().__init__(data, title=title, subject_id_name=subject_id_name, meta={}, is_plot_subgraph=False)
        self._idv_name = idv_name
        self._dv_name = dv_name
        self._x_scale: Literal["linear", "log"] = x_scale
        self._y_scale: Literal["linear", "log"] = y_scale

    def _make_plot(self) -> go.Figure:
        fig = go.Figure()
        if self._id_name:
            for _id, sub_df in self._data.groupby(self._id_name):
                single_curve = _individual_curve(x=sub_df[self._idv_name], y=sub_df[self._dv_name], name=str(_id))
                fig.add_trace(single_curve)
        else:
            curve = _individual_curve(x=self._data[self._idv_name], y=self._data[self._dv_name], name="")
            fig.add_trace(curve)

        fig.update_xaxes(
            title={'text': self._idv_name},
            type=self._x_scale,
        )
        fig.update_yaxes(
            title={
                'text': self._dv_name, 'standoff': 0
            }, type=self._y_scale, exponentformat='e', showexponent='all'
        )
        fig.update_layout(height=600, width=800, title={
            "text": self._title, 'y': 0.92, 'x': 0.5
        })
        return fig

    def _create_self(
        self,
        data: pd.DataFrame,
        title: str,
        subject_id_name: str | None,
        meta: dict[str, str],
        is_plot_subgraph: bool = False
    ) -> 'SimulateIndividualPlot':
        return SimulateIndividualPlot(
            data,
            subject_id_name=subject_id_name,
            idv_name=self._idv_name,
            dv_name=self._dv_name,
            title=title,
            x_scale=self._x_scale,
            y_scale=self._y_scale
        )




[docs]class SimulateIntervalPlot(AbstractGroupbyablePlotlyPlot):
    percents = [0.05, 0.15, 0.25, 0.35, 0.45, 0.55, 0.65, 0.75, 0.85, 0.95, 0.5]
    fill_colors = [
        'rgba(11,122,111,0.3)',
        'rgba(11,122,99,0.4)',
        'rgba(11,122,77,0.6)',
        'rgba(11,122,55,0.7)',
        'rgba(11,122,44,0.8)',
        'rgba(11,122,55,0.7)',
        'rgba(11,122,77,0.6)',
        'rgba(11,122,99,0.4)',
        'rgba(11,122,111,0.3)'
    ]

    def __init__(
        self,
        data: pd.DataFrame,
        subject_id_name: str | None,
        idv_name: str,
        dv_name: str,
        title: str = "",
        x_scale: Literal["linear", "log"] = "linear",
        y_scale: Literal["linear", "log"] = "linear",
    ) -> None:
        super().__init__(data, title=title, subject_id_name=subject_id_name, meta={}, is_plot_subgraph=False)
        self._idv_name = idv_name
        self._dv_name = dv_name
        self._x_scale: Literal["linear", "log"] = x_scale
        self._y_scale: Literal["linear", "log"] = y_scale

    def _process_data(self):

        def compute_quantiles(df: pd.DataFrame) -> pd.DataFrame:
            percents_by_time = df[self._dv_name].quantile(self.percents)
            return percents_by_time.to_frame().T

        results: pd.DataFrame = self._data.groupby(self._idv_name).apply(compute_quantiles)
        results = results.reset_index()
        return results

    def _make_plot(self) -> go.Figure:
        fig = go.Figure()

        # percents = ['%.03f' % (_ * 0.1 + 0.05) for _ in range(10)]
        results = self._process_data()
        idv: npt.NDArray[np.float_] = results[self._idv_name].to_numpy()
        x = np.concatenate([idv, idv[::-1]])
        for i in range(len(self.percents) - 2):
            line1: npt.NDArray[np.float_] = results[self.percents[i]].to_numpy()
            line2: npt.NDArray[np.float_] = results[self.percents[i + 1]].to_numpy()
            y = np.concatenate([line1, line2[::-1]])
            # 分布图
            fig.add_trace(
                go.Scatter(
                    x=x,
                    y=y,
                    fill='toself',
                    fillcolor=self.fill_colors[i],
                    line={
                        'color': 'rgba(255,255,255)', 'width': 0
                    },
                    showlegend=False,
                    name=f'area:{i}',
                    text=f'{5 + i * 10}%-{15 + i * 10}%',
                    hoveron='fills',
                    hoverinfo='text',
                )
            )

        # 中位线
        fig.add_trace(
            go.Scatter(
                x=idv,
                y=results[0.5],
                mode='lines',
                name='median line',
                line=dict(color='green', width=1),
                hoverinfo='skip',
                showlegend=False,
            ),
        )

        fig.update_xaxes(title={'text': self._idv_name}, type=self._x_scale)
        fig.update_yaxes(
            title={
                'text': self._dv_name, 'standoff': 0
            }, type=self._y_scale, exponentformat='e', showexponent='all'
        )
        fig.update_layout(height=600, width=800, title={
            "text": self._title, 'y': 0.92, 'x': 0.5
        })
        return fig
