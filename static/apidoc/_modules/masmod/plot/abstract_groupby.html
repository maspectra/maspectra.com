[docs]    def groupby(self: SelfType, by: str | list[str]) -> list[SelfType]:
        """
        Groupby data and plot

        Args:
            by: The columns name used to groupby the data, should be type str or list of str

        Returns:
            A list of the same graph objects
        """
        group_name: str
        is_subplot: bool = self._is_subplot  # 先继承这一层是否需要绘制子图
        meta: dict[str, str] = deepcopy(self._meta)  # 先继承这一层存储的分类信息
        if (isinstance(by, str)):  # 对入参合法性校验
            group_name = by
            if (group_name not in list(self._data.columns)):
                raise ValueError("{} column is not in the data.".format(group_name))
            if (self._is_subplot) and (self._id_name == by):  # 需要画子图但传入的 id 列已经被 group by 时不绘制子图
                is_subplot = False
        elif (isinstance(by, list)):
            for group_name in by:
                if (group_name not in list(self._data.columns)):
                    raise ValueError("{} column is not in the data.".format(group_name))
            if (self._is_subplot) and (self._id_name in by):  # 需要画子图但传入的 id 列已经被 group by 时不绘制子图
                is_subplot = False
        else:
            raise NotImplementedError("parameter 'by' has not supported  type {} yet.".format(type(by)))

        group_list: list[SelfType] = []  # 创建返回的实例列表
        for group_key, group_data in self._data.groupby(by, sort=False):

            new_meta: dict[str, str] = {}
            new_title_str: str = " "  # 存一下标题字符串
            if (isinstance(by, str)):  # 当只有一个分组列时对 str 和 str 进行操作，这里 group_key 的类型不定无法判断
                new_meta[str(by)] = str(group_key)
                new_title_str += str(by) + " = " + str(group_key)
            elif (isinstance(group_key, tuple)
                 ) and (isinstance(by, list)) and (len(by) == len(group_key)):  # 当只有不只一个分组列时对 list 和 tuple 进行操作
                for i in range(len(by)):
                    new_meta[str(by[i])] = str(group_key[i])
                    new_title_str += str(by[i]) + " = " + str(group_key[i]) + " "
            else:
                raise RuntimeError("'groupby' failed, please check the type of parameter.")

            meta.update(new_meta)  # 更新子图的 meta，如果不需要把之前的信息继承，则这里不要 update
            title_str: str = self._title + new_title_str
            group_list.append(
                self._create_self(
                    data=group_data,
                    title=title_str,
                    subject_id_name=self._id_name,
                    meta=meta,
                    is_plot_subgraph=is_subplot
                )
            )

        return group_list
