    def _auto_bin(self, bin: bool | int | tuple[int, int]) -> list[list[float]]:
        est_data: pd.DataFrame = self._data[self._data[COL_SIMULATE] == 0]
        idv_values: list[float] = est_data[self.__idv_name].to_numpy().tolist()
        idv_values = sorted(idv_values)
        unique_idv: list[float] = np.unique(idv_values).tolist()

        if (isinstance(bin, bool)) and (bin == True):
            if len(unique_idv) < 2:
                raise ValueError("Too few idv values.")
            range_max: int = 20 if len(unique_idv) >= 20 else len(unique_idv)
            idv_group: list[
                list[float]
            ] = auto_bin_api(unique_idv, idv_values, range_min=2, range_max=range_max, min_points=self.__min_points)
        elif (isinstance(bin, int)):
            if (bin > 1):
                if len(unique_idv) < bin:
                    raise ValueError("Too few idv points compared to the number of bins.")
                idv_group: list[
                    list[float]
                ] = auto_bin_api(unique_idv, idv_values, range_min=bin, range_max=bin, min_points=self.__min_points)
            else:
                raise ValueError("When defining the number of bins, it should be greater than 1.")
        elif (isinstance(bin, tuple)):
            if (bin[0] <= 1):
                raise ValueError("When defining a range of bins, the minimum value should be greater than 1.")
            elif (bin[0] >= bin[1]):
                raise ValueError(
                    "When defining a range of bins, the maximum value should be greater than the minimum value."
                )
            elif (bin[1] >= len(unique_idv)):
                if len(unique_idv) < bin[0]:
                    raise ValueError("Too few idv points compared to the number of min bins.")
                idv_group: list[list[float]] = auto_bin_api(
                    unique_idv, idv_values, range_min=bin[0], range_max=len(unique_idv), min_points=self.__min_points
                )
            else:
                if len(unique_idv) < bin[0]:
                    raise ValueError("Too few idv points compared to the number of min bins.")
                idv_group: list[list[float]] = auto_bin_api(
                    unique_idv, idv_values, range_min=bin[0], range_max=bin[1], min_points=self.__min_points
                )
        else:
            raise ValueError("Define a error type of bin.")

        return (idv_group)

    def _compute_ci(self) -> tuple[pd.DataFrame, pd.DataFrame]:
        BIN_MEDIAN_IDV: str = "bin_median_idv"
        extend_data: pd.DataFrame = self._data

        def get_median_idv(idv_list: list[float]):
            return np.repeat(np.median(idv_list), len(idv_list)).tolist()

        median_idv: list[float] = sum(list(map(get_median_idv, self.__idv_group)), [])  # 生成每个 bin 值和中位数的字典
        median_df = pd.DataFrame({
            self.__idv_name: sum(self.__idv_group, []), BIN_MEDIAN_IDV: median_idv
        })
        extend_data = extend_data.merge(median_df, on=[self.__idv_name], how="left")

        obs_data: pd.DataFrame = extend_data[extend_data[COL_SIMULATE] == 0]
        simu_data: pd.DataFrame = extend_data[extend_data[COL_SIMULATE] != 0]

        obs_percentile: pd.DataFrame = pd.DataFrame()
        obs_percentile["idv"] = np.unique(obs_data[BIN_MEDIAN_IDV].to_numpy())
        obs_percentile["obs_l"] = obs_data.groupby([BIN_MEDIAN_IDV]
                                                  )[self.__dv_name].quantile(self.__ci_range_lower).values
        obs_percentile["obs_m"] = obs_data.groupby([BIN_MEDIAN_IDV])[self.__dv_name].median().values
        obs_percentile["obs_h"] = obs_data.groupby([BIN_MEDIAN_IDV]
                                                  )[self.__dv_name].quantile(self.__ci_range_upper).values

        simu_percentile: pd.DataFrame = pd.DataFrame()
        simu_percentile["idv"] = np.unique(simu_data[BIN_MEDIAN_IDV].to_numpy())
        simu_percentile_l: MasSeries = typing.cast(
            MasSeries,
            simu_data.groupby([BIN_MEDIAN_IDV, COL_SIMULATE])[COL_SIMULATE_VALUE].quantile(self.__ci_range_lower)
        )
        simu_percentile["ll"] = simu_percentile_l.groupby(BIN_MEDIAN_IDV).quantile(self.__ci_range_lower).values
        simu_percentile["lm"] = simu_percentile_l.groupby(BIN_MEDIAN_IDV).median().values
        simu_percentile["lh"] = simu_percentile_l.groupby(BIN_MEDIAN_IDV).quantile(self.__ci_range_upper).values
        simu_percentile_m: MasSeries = typing.cast(
            MasSeries, simu_data.groupby([BIN_MEDIAN_IDV, COL_SIMULATE])[COL_SIMULATE_VALUE].median()
        )
        simu_percentile["ml"] = simu_percentile_m.groupby(BIN_MEDIAN_IDV).quantile(self.__ci_range_lower).values
        simu_percentile["mm"] = simu_percentile_m.groupby(BIN_MEDIAN_IDV).median().values
        simu_percentile["mh"] = simu_percentile_m.groupby(BIN_MEDIAN_IDV).quantile(self.__ci_range_upper).values
        simu_percentile_h: MasSeries = typing.cast(
            MasSeries,
            simu_data.groupby([BIN_MEDIAN_IDV, COL_SIMULATE])[COL_SIMULATE_VALUE].quantile(self.__ci_range_upper)
        )
        simu_percentile["hl"] = simu_percentile_h.groupby(BIN_MEDIAN_IDV).quantile(self.__ci_range_lower).values
        simu_percentile["hm"] = simu_percentile_h.groupby(BIN_MEDIAN_IDV).median().values
        simu_percentile["hh"] = simu_percentile_h.groupby(BIN_MEDIAN_IDV).quantile(self.__ci_range_upper).values

        obs_percentile.sort_values(by="idv", axis=0, ascending=True, inplace=True)
        simu_percentile.sort_values(by="idv", axis=0, ascending=True, inplace=True)
        return obs_percentile.reset_index(drop=True), simu_percentile.reset_index(drop=True)

def _parallel_simulate(
    result: FitResult, samples: int, idv_col: str, dv_col: str, subject_col: str | None
) -> pd.DataFrame:
    TEMP_ID: str = "temp_id"
    if subject_col is None:
        id_list: npt.NDArray[np.float_] = np.array([0] * result._fit_data.shape[0])
        n_id: int = 1
    else:
        n_id: int = result._fit_data[subject_col].nunique()
        id_list: npt.NDArray[np.float_] = np.repeat(
            range(n_id), result._fit_data[subject_col].value_counts(sort=False).to_numpy().tolist()
        )  # 将 temp_id 依照每个个体对应的事件数量进行扩展，使其长度和数据的长度相同

    simulate_id: npt.NDArray[np.float_] = np.repeat(range(1, samples + 1), result._fit_data.shape[0])  # 生成模拟次数的ID
    simulate_temp_id: npt.NDArray[np.float_
                                 ] = np.tile(id_list,
                                                (1, samples))[0] + (simulate_id - 1) * n_id  # 生成模拟后用于拟合的 temp_id
    simulate_df: pd.DataFrame = pd.concat(
        [
            pd.DataFrame({
                TEMP_ID: simulate_temp_id, COL_SIMULATE: simulate_id
            }),
            pd.DataFrame(np.tile(result._fit_data, (samples, 1)), columns=result._fit_data.columns)  # 扩展原数据进行合并
        ],
        axis=1
    ).reset_index(drop=True)

    simulate_param_dict: typing.Dict[str, float | Theta | Eta | Eps] = _param_to_dict(result.final_pop_params)

    # import time
    # start = time.time()
    simulate_data = ModelData(simulate_df, id=TEMP_ID, time=idv_col, dv=dv_col)
    sim_result = simulate(
        mod=result._mod, model_ptr=result._model_ptr, data=simulate_data, parameters=simulate_param_dict
    )
    simu_result_data: pd.DataFrame = sim_result.simulated_df[[COL_SIMULATE] + list(result._fit_data.columns) +

    # print("C++: ", time.time() - start)

    est_data_obs[COL_SIMULATE] = 0
    est_data_obs[COL_SIMULATE_VALUE] = 0

def _param_to_dict(param: PopParams) -> typing.Dict[str, float | Theta | Eta | Eps]:
    param_dict: typing.Dict[str, float | Theta | Eta | Eps] = dict()
    param_dataframe: pd.DataFrame = param.as_pandas()
    for i in param.theta_names:
        param_dict[i] = theta(param_dataframe[i].to_numpy()[0])
    for i in param.eta_names:
        param_dict[i] = omega(param_dataframe[i].to_numpy()[0])
    for i in param.epsilon_names:
        param_dict[i] = sigma(param_dataframe[i].to_numpy()[0])
    return param_dict
