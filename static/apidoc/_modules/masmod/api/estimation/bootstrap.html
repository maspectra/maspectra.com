    def __init__(

        self.__result = fit_result

        self.__elapsed_time = elapsed_time

        self.__param_names = self.__result.final_pop_params.theta_names + self.__result.final_pop_params.eta_names + self.__result.final_pop_params.epsilon_names

        if bootstrap_df is None:
            self.__bootstrap_df = self.__generate_df()
        else:
            self.__bootstrap_df = bootstrap_df

    def __repr__(self) -> str:
        return str(self)

    def __str__(self) -> str:
        tol: int = 3
        param_median: list[float] = []
        param_ci: list[str] = []
        for param_name in self.__param_names:
            arr: npt.NDArray[np.float_] = self.__bootstrap_df[param_name].to_numpy()
            param_median.append(round(np.median(arr, axis=0), tol))
            param_ci.append(
                f"{np.percentile(arr, ((1 - self.__ci_range) / 2) * 100, axis=0):.3f} - {np.percentile(arr, (1 - (1 - self.__ci_range) / 2) * 100, axis=0):.3f}"
            )

        headers = ["Parameter", "Estimate", "RSE(%)", "Median", str(self.__ci_range * 100) + "% CI"]
        n_theta = len(self.__result.final_pop_params.theta)
        n_eta = len(self.__result.final_pop_params.omega)
        params_tabular_data = self.__result.final_pop_params.as_tabular_data(format=False)
        _tabular_data: list[tuple[str | int | float, ...]] = [() for _ in range(len(params_tabular_data))]
        rse_diag = list(self.__result.rse)

        for i in range(len(params_tabular_data)):
            if i == 0 or \
                  i == n_theta + 1 or \
                  i == n_theta + n_eta + 2:
                # Theta / Omega / Sigma Title
                _tabular_data[i] = (params_tabular_data[i][0], "", "", "", "")
            else:
                rse_diag_i = rse_diag.pop(0)
                param_median_i = param_median.pop(0)
                param_ci_i = param_ci.pop(0)

                _tabular_data[i] = (
                    params_tabular_data[i][0],
                    f"{params_tabular_data[i][1]:.3f}",
                    f"{rse_diag_i:.1f}",
                    f"{param_median_i:.3f}",
                    param_ci_i
                )

        meta_summary = tabulate(
            # tabular_data={
            #     "Parameter": self.__param_names,
            #     "Estimate": estimate_param.tolist(),
            #     "RSE(%)": self.__est.rse.tolist(),
            #     "Median": param_median, (str(self.__ci_range * 100) + "% CI"): param_ci
            # },
            tabular_data=_tabular_data,
            headers=headers,
            tablefmt="github",
            numalign="right",
            stralign="left",
            colalign=["left", "left", "left", "left"]
        )

        return "Successful Rate: {}%\nElapsed Time: {}\n{}".format(
            np.sum(self.__bootstrap_df["Successful"].to_numpy(), axis=0) / len(self.__bs_list) * 100,
            self.__elapsed_time,
            meta_summary
        )

    def __generate_df(self) -> pd.DataFrame:
        name_base: list[str] = ["Bootstrap Model", "Successful", "Objective Function Value"]
        name_param: list[str] = self.__param_names
        name_param_se: list[str] = []
        for i in name_param:
            name_param_se.append("se_" + i)
        col_names = name_base + name_param + name_param_se

        bootstrap_df: pd.DataFrame = pd.DataFrame(columns=col_names)
        bootstrap_count: int = 1
        for est in self.__bs_list:
            bootstrap_i: dict[str, list[float]] = {}
            bootstrap_i["Bootstrap Model"] = [bootstrap_count]
            bootstrap_i["Successful"] = [int(est.converged)]
            bootstrap_i["Objective Function Value"] = [est.final_obj_val]
            param_index: int = 0
            for theta_index in range(len(est.final_pop_params.theta)):
                bootstrap_i[name_param[param_index]] = [est.final_pop_params.theta[theta_index]]
                param_index += 1
            for omega_index in range(len(est.final_pop_params.omega)):
                bootstrap_i[name_param[param_index]] = [est.final_pop_params.omega[omega_index][omega_index]]
                param_index += 1
            for sigma_index in range(len(est.final_pop_params.sigma)):
                bootstrap_i[name_param[param_index]] = [est.final_pop_params.sigma[sigma_index][sigma_index]]
                param_index += 1
            for se_index in range(len(name_param_se)):
                bootstrap_i[name_param_se[se_index]] = [est.compute_se()[se_index][se_index]]

            if len(bootstrap_df.columns) == len(bootstrap_i):
                bootstrap_df = pd.concat([bootstrap_df, pd.DataFrame(bootstrap_i)], axis=0)
            else:
                raise ValueError("An error occurred in the generation of bootstrap table.")
            bootstrap_count += 1

        return bootstrap_df

def _bootstrap_sampling(data: ModelData, samples: int, seed: int | None = None) -> list[pd.DataFrame]:
    if seed is not None:
        np.random.seed(seed)

    if (data.reserved_colnames.get("subject") is None):
        raise IndexError("Bootstrap needs subject columns, but not given")
    else:
        subject_list: list[pd.DataFrame] = []
        for _, subject_data in data.groupby(str(data.reserved_colnames.get("subject")), sort=False):
            subject_list.append(subject_data.reset_index(drop=True))

    data_list: list[pd.DataFrame] = []
    for _ in range(samples):
        sample_i: list[int] = sorted(
            list(np.random.choice(range(len(subject_list)), size=len(subject_list), replace=True))
        )
        data_i: pd.DataFrame = pd.DataFrame(columns=[BOOTSTRAP_ID_COLUMN] + list(data.columns))
        for i in range(len(sample_i)):
            bootstrap_id: pd.DataFrame = pd.DataFrame(
                {
                    BOOTSTRAP_ID_COLUMN: [i + 1] * subject_list[sample_i[i]].shape[0]
                }
            ).reset_index(drop=True)
            subject_i_data: pd.DataFrame = pd.concat([bootstrap_id, subject_list[sample_i[i]]], axis=1)
            data_types = data.dtypes
            data_types[BOOTSTRAP_ID_COLUMN] = "int64"
            data_i = pd.concat([data_i, subject_i_data], axis=0).astype(data_types)
        data_list.append(data_i)

    return data_list



[docs]def bootstrap(
    fit_result: FitResult,
    samples: int,
    seed: int | None = None,
) -> BootstrapResult:
    """
        Run bootstrap

        Args:
            fit_result: The result of the model fit
            model: The population model
            samples: The number of bootstraps
            seed: Random seed

        Returns:
            Bootstrap results
    """
    time_stamp = datetime.datetime.now()
    bootstrap_data_list: list[pd.DataFrame
                             ] = _bootstrap_sampling(data=fit_result._fit_data, samples=samples, seed=seed)
    result_list: list[FitResult] = []

    for i in range(samples):
        print(f"➡️ Bootstrap #{i+1}\n\n")
        data_i = ModelData(bootstrap_data_list[i].reset_index(drop=True), id=BOOTSTRAP_ID_COLUMN)
        check_model_data(data_i, fit_result._mod)
        fit_args = {
            **fit_result._fit_args
        }
        if "log_level" in fit_args.keys():
            fit_args["log_level"] = 0

        result_i: FitResult = _fit(
            data=data_i,
            init_pop_params=fit_result._init_pop_params,
            mod=fit_result._mod,
            estimator=fit_result._estimator,
            ptr=fit_result._model_ptr,
            **fit_args
        )
        result_i.compute_se()
        result_list.append(result_i)

    elapsed_time = datetime.datetime.now() - time_stamp
    return BootstrapResult(
        fit_result=fit_result, bootstrap_result_list=result_list, elapsed_time=elapsed_time, ci_range=0.95
    )
