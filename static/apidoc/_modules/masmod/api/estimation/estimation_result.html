    def __init__(
        self,
        model_ptr: DittoPopModel,
        mod: InterpretedModule,
        ditto_est_results: DittoEstimationResult,
        elapsed_time: datetime.timedelta,
        previous_est_result: EstimationResult | None = None,
    ) -> None:
        self.__model_ptr = model_ptr
        self.__mod = mod
        self._ptr = ditto_est_results
        self.__final_pop_params = PopParams.from_ditto(mod=self.__mod, ditto_pop_params=self._ptr.final_pop_params)
        self.__elapsed_time = elapsed_time

        self._previous_est_result = previous_est_result

        # ebe 缓存
        self.__ebe: list[npt.NDArray[np.float_]] | None = None

        # shrinkage 缓存
        self.__shrinkage: npt.NDArray[np.float_] | None = None

    def __repr__(self) -> str:
        return str(self)

    def __str__(self) -> str:
        fit_meta_headers, fit_meta_data = self._meta_as_tabular_data()
        est_headers, est_data = self._estimation_as_tabular_data(colorful=False)

        meta_summary = tabulate(
            tabular_data=fit_meta_data,
            headers=fit_meta_headers,
            tablefmt="github",
            colalign=["left", "left", "left", "left"]
        )

        params_summary = tabulate(
            tabular_data=est_data, headers=est_headers, tablefmt="github", colalign=["left", "left", "left", "left"]
        )

        out = f"""================= Summary =================

*** Estimation Summary ***
{meta_summary}

*** Parameter Estimation ***
{params_summary}
"""

        return out

    def _ipython_display_(self) -> None:
        from IPython.display import display_markdown
        fit_meta_headers, fit_meta_data = self._meta_as_tabular_data()
        est_headers, est_data = self._estimation_as_tabular_data(colorful=True)

        display_markdown("#### Estimation Summary", raw=True)

        meta_summary = tabulate(
            tabular_data=fit_meta_data,
            headers=fit_meta_headers,
            tablefmt="pipe",
            numalign="right",
            stralign="left",
            colalign=["left", "left", "left", "left"]
        )
        display_markdown(meta_summary, raw=True)

        display_markdown("\n#### Parameter Estimation", raw=True)

        params_summary = tabulate(
            tabular_data=est_data,
            headers=est_headers,
            tablefmt="pipe",
            numalign="right",
            stralign="left",
            colalign=["left", "left", "left", "left"]
        )
        display_markdown(params_summary, raw=True)

    @property
    def empirical_bayesian_estimation(self) -> list[npt.NDArray[np.float_]]:
        """Empirical Bayesian Estimation (eta_hat)"""
        if self.__ebe is None:
            self.__ebe = empirical_bayesian_estimation(pop_params=self.final_pop_params, model=self.__model_ptr)
        return self.__ebe

    @property
    def shrinkage(self) -> npt.NDArray[np.float_]:
        if self.__shrinkage is None:
            ebe = self.empirical_bayesian_estimation
            eta_std = np.std(np.array(ebe), axis=0)
            omega_shrinkage = np.ones(len(self.final_pop_params.omega)
                                     ) - np.divide(eta_std, np.sqrt(np.diag(self.final_pop_params.omega)))

            sigma_shrinkage: npt.NDArray[np.float_] = np.zeros(len(self.final_pop_params.sigma), dtype=float)
            for i in range(len(self.final_pop_params.sigma)):
                sigma_shrinkage[i] = compute_shrinkage_sigma(
                    eta_ebe_sample=ebe, pop_params=self.final_pop_params, model=self.__model_ptr, sigma_index=i
                )

            self.__shrinkage = np.array([*omega_shrinkage, *sigma_shrinkage], dtype=float)
        return self.__shrinkage

    def _meta_as_tabular_data(self) -> tuple[list[str], list[tuple[str | int | float, ...]]]:
        fmt_converged = str(self.converged)
        if self.converged:
            fmt_converged += " ✅"
        else:
            fmt_converged += " ❌"

        fmt_elapsed_time = format_timedelta(self.elapsed_time)
        return ["Final Objective Value", "Converged", "CPU Elapsed Time",
                "Iterations"], [(self.final_obj_val, fmt_converged, fmt_elapsed_time, self.n_iter)]

    def _estimation_as_tabular_data(self, colorful: bool) -> tuple[list[str], list[tuple[str | int | float, ...]]]:
        n_theta = len(self.__mod.thetas)
        n_eta = len(self.__mod.etas)
        # n_eps = len(self.__mod.epsilons)
        params_tabular_data = self.final_pop_params.as_tabular_data(format=False)
        _tabular_data: list[tuple[str | int | float, ...]] = [() for _ in range(len(params_tabular_data))]
        shrinkage: list[float] = list(self.shrinkage)

        for i in range(len(params_tabular_data)):
            if i == 0 or \
                  i == n_theta + 1 or \
                  i == n_theta + n_eta + 2:
                # Theta / Omega / Sigma Title
                _tabular_data[i] = (params_tabular_data[i][0], "", "", "")
            else:
                fmt_shrinkage: str = "-"
                # Theta does not have any shrinkage
                if i > n_theta + 1:
                    # self.__shrinkage starts with omega
                    shrinkage_i = shrinkage.pop(0)
                    shrinkage_value: float = round(shrinkage_i, 3) * 100
                    fmt_shrinkage = f"{shrinkage_value:.1f}"

                    if colorful and shrinkage_value > 30:
                        fmt_shrinkage = f"<span style='color: #dba856'>{fmt_shrinkage}</span>"  # yellow like

                _tabular_data[i] = (params_tabular_data[i][0], f"{params_tabular_data[i][1]:.3f}", fmt_shrinkage)

        return ["Parameter", "Estimate", "Shrinkage(%)"], _tabular_data
