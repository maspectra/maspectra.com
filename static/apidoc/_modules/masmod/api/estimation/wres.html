    def __init__(
        self, est_result: DittoEstimationResult, model_ptr: DittoPopModel, method: DittoEstimationMethod
    ) -> None:
        self.ditto_wres_compute: DittoWresCompute = DittoWresCompute(
            pop_params=est_result.final_pop_params, model=model_ptr
        )
        self.method = method
        self.wres: list[npt.NDArray[np.float_]] | None = None
        self.cwres: list[npt.NDArray[np.float_]] | None = None
        self.cwresi: list[npt.NDArray[np.float_]] | None = None
        self.iwres: list[npt.NDArray[np.float_]] | None = None
        self.ciwres: list[npt.NDArray[np.float_]] | None = None
        self.ciwresi: list[npt.NDArray[np.float_]] | None = None

        self, option: typing.Literal["WRES", "CWRES", "CWRESI", "IWRES", "CIWRES", "CIWRESI"]
    ) -> list[npt.NDArray[np.float_]]:
        if (option == "WRES"):
            if (self.wres is None):
                if (self.method == DittoEstimationMethod.ESTIMATION_METHOD_FO):
                    self.wres = self.ditto_wres_compute.compute_fo_wres()
                elif (self.method == DittoEstimationMethod.ESTIMATION_METHOD_FOCE):
                    self.wres = self.ditto_wres_compute.compute_focei_wres()
                elif (self.method == DittoEstimationMethod.ESTIMATION_METHOD_SAEM):
                    self.wres = self.ditto_wres_compute.compute_fo_wres()  #TODO 待补充

                    raise ValueError("This fitting method is not supported temporarily.")

[docs]    def compute_wres_series(
        self, option: typing.Literal["WRES", "CWRES", "CWRESI", "IWRES", "CIWRES", "CIWRESI"]
    ) -> pd.Series:
        joined: npt.NDArray[np.float_] = np.array([], dtype=float)
        wres_results: list[npt.NDArray[np.float_]] = self.compute_wres_list(option=option)
        for ind_wres in wres_results:
            joined = np.append(joined, ind_wres)
        series = pd.Series(joined, name=option)
        return series
