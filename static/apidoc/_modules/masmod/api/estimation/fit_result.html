    def __init__(
        self,
        model_ptr: DittoPopModel,
        data: ModelData,
        mod: InterpretedModule,
        estimator: DittoEstimator,
        est_results: tuple[EstimationResult, ...],
        init_pop_params: PopParams,
        fit_args: dict[str, typing.Any],
    ) -> None:
        self._model_ptr = model_ptr
        self.__data = data.copy(deep=True)
        self._mod = deepcopy(mod)
        self._estimator = estimator
        self.__est_results = est_results
        self._init_pop_params = init_pop_params
        if len(est_results) == 0:
            raise ValueError("Length of 'est_results' must be greater or equal to 1")

        self.__final_est_result = est_results[-1]

        self.__final_pop_params = PopParams.from_ditto(
            mod=self._mod, ditto_pop_params=self.__final_est_result.final_pop_params
        )

        def timedelta_reducer(t1: datetime.timedelta, t2: datetime.timedelta) -> datetime.timedelta:
            return t1 + t2

        self.__ttl_elapsed_time: datetime.timedelta = reduce(timedelta_reducer, [e.elapsed_time for e in est_results])
        self.__fit_args = fit_args

        # ebe 缓存
        self.__ebe: list[npt.NDArray[np.float_]] | None = None

        # shrinkage 缓存
        self.__shrinkage: npt.NDArray[np.float_] | None = None

        # se 缓存
        # tuple[ComputeSEArgsType, SENumpy]
        self.__se: tuple[tuple[typing.Literal["S", "R", "RS"] | typing.Literal["LINEAR", "SA"] | None,
                                DittoSASESetting | None],
                            npt.NDArray[np.float_]] | None = None

        # pred and ipred 缓存
        self.__predictions: _Predictions | None = None

        # wres and iwres 缓存
        self.__wres: Wres = Wres(self.__final_est_result._ptr, model_ptr, self.__final_est_result._ptr.est_method)

    @property
    def rse(self) -> np.ndarray[typing.Literal[1], np.dtype[np.float_]]:
        """Final standard error for parameters"""
        n_theta = len(self._mod.thetas)
        n_eta = len(self._mod.etas)
        n_eps = len(self._mod.epsilons)
        n_params = n_theta + n_eta + n_eps
        rse = np.zeros(n_params)
        if self.__se is None:
            self.compute_se()
        if self.__se is not None:
            _, se_2d = self.__se
            se_diag = list(np.sqrt(np.diag(se_2d)))
            params_tabular_data = self.final_pop_params.as_tabular_data(format=False)

            param_index = 0
            for i in range(len(params_tabular_data)):
                if i == 0 or \
                  i == n_theta + 1 or \
                  i == n_theta + n_eta + 2:
                    # Theta / Omega / Sigma Title
                    continue
                se_diag_i = se_diag.pop(0)
                rse[param_index] = round(se_diag_i /
                                            params_tabular_data[i][1], 3) * 100 if (se_diag is not None) else "-"
                param_index += 1
        return rse

    @property
    def shrinkage(self) -> npt.NDArray[np.float_]:
        if self.__shrinkage is None:
            ebe = self.empirical_bayesian_estimation
            eta_std = np.std(np.array(ebe), axis=0)
            omega_shrinkage = np.ones(len(self.final_pop_params.omega)
                                     ) - np.divide(eta_std, np.sqrt(np.diag(self.final_pop_params.omega)))

            sigma_shrinkage: npt.NDArray[np.float_] = np.zeros(len(self.final_pop_params.sigma), dtype=float)
            for i in range(len(self.final_pop_params.sigma)):
                sigma_shrinkage[i] = compute_shrinkage_sigma(
                    eta_ebe_sample=ebe, pop_params=self.final_pop_params, model=self._model_ptr, sigma_index=i
                )

            self.__shrinkage = np.array([*omega_shrinkage, *sigma_shrinkage], dtype=float)
        return self.__shrinkage

        )
        ipred_arr = np.array([], dtype=float)
        for ind_ipred in ipred:
            ipred_arr = np.append(ipred_arr, ind_ipred)

        # #TODO @yujiesen: 临时修复，后续优化

        #     data_drop_dv_nan = self.__data[~dv_nan_index].reset_index(drop=True)
        # else:
        #     raise ValueError('EstimationResult must have data')

        if "mdv" in self.__data.kwd_mapping:
            mdv_column = self.__data.kwd_mapping["mdv"]
            concate_data = self.__data.sampling.reset_index(drop=True)

[docs]    def compute_se(
        self,
        se_option: typing.Literal["S", "R", "RS"] | typing.Literal["LINEAR", "SA"] | None = None,
        saem_se_settings: DittoSASESetting | None = None
    ) -> npt.NDArray[np.float_]:
        if self.__se is not None:
            cached_args, cached_result = self.__se
            cached_se_option, cached_saem_se_settings = cached_args
            if cached_se_option == se_option and cached_saem_se_settings == saem_se_settings:
                return cached_result

        est_method = self.__final_est_result._ptr.est_method

        if est_method == DittoEstimationMethod.ESTIMATION_METHOD_SAEM:
            res = self.__do_compute_se_saem(se_option, saem_se_settings)
        else:
            if saem_se_settings is not None:
                raise ValueError("SAEM SE settings should not be given in Laplacian SE calculation")
            res = self.__do_compute_se_laplacian(se_option)

        self.__se = ((se_option, saem_se_settings), res)

        return res


    def _fit_meta_as_tabular_data(self) -> tuple[list[str], list[tuple[str | int | float, ...]]]:

    def _estimation_as_tabular_data(self, colorful: bool) -> tuple[list[str], list[tuple[str | int | float, ...]]]:
        n_theta = len(self._mod.thetas)
        n_eta = len(self._mod.etas)
        n_eps = len(self._mod.epsilons)
        params_tabular_data = self.final_pop_params.as_tabular_data(format=False)
        _tabular_data: list[tuple[str | int | float, ...]] = [() for _ in range(len(params_tabular_data))]
        se_diag: list[float] | None = None
        shrinkage: list[float] = list(self.shrinkage)
        if self.__se is not None:
            _, se_2d = self.__se
            # se_diag = list(np.sqrt(np.diag(se_2d)))
            se_diag = list(np.diag(se_2d))
            j = 2
            for i in range(n_eta + n_eps):
                if (i == n_eta):
                    j += 1
                se_diag[n_theta + i] = se_diag[n_theta + i] / (2 * params_tabular_data[n_theta + i + j][1])

        for i in range(len(params_tabular_data)):
            if i == 0 or \
                  i == n_theta + 1 or \
                  i == n_theta + n_eta + 2:
                # Theta / Omega / Sigma Title
                _tabular_data[i] = (params_tabular_data[i][0], "", "", "")
            else:
                fmt_se = "-"
                if se_diag:
                    se_diag_i = se_diag.pop(0)
                    se_value: float = round(se_diag_i / params_tabular_data[i][1], 3) * 100
                    fmt_se: str = f"{se_value:.1f}"
                    if colorful:
                        if se_value > 100:
                            fmt_se = f"<span style='color: #d82020'>{fmt_se}</span>"  # red like
                        elif se_value > 50:
                            fmt_se = f"<span style='color: #dba856'>{fmt_se}</span>"  # yellow like

                fmt_shrinkage: str = "-"
                # Theta does not have any shrinkage
                if i > n_theta + 1:
                    # self.__shrinkage starts with omega
                    shrinkage_i = shrinkage.pop(0)
                    shrinkage_value: float = round(shrinkage_i, 3) * 100
                    fmt_shrinkage = f"{shrinkage_value:.1f}"

                    if colorful and shrinkage_value > 30:
                        fmt_shrinkage = f"<span style='color: #dba856'>{fmt_shrinkage}</span>"  # yellow like

                _tabular_data[i] = (
                    params_tabular_data[i][0], f"{params_tabular_data[i][1]:.3f}", fmt_se, fmt_shrinkage
                )

        return ["Parameter", "Estimate", "RSE(%)", "Shrinkage(%)"], _tabular_data

    def __do_compute_se_laplacian(
        self,
        se_option: typing.Literal["S", "R", "RS"] | typing.Literal["LINEAR", "SA"] | None = None
    ) -> npt.NDArray[np.float_]:
        ditto_se_option: DittoSEOption
        if se_option == "S":
            ditto_se_option = DittoSEOption.DITTO_SE_OPTION_S
        elif se_option == "R":
            ditto_se_option = DittoSEOption.DITTO_SE_OPTION_R
        elif se_option == "RS":
            ditto_se_option = DittoSEOption.DITTO_SE_OPTION_RS
        elif se_option is None:
            ditto_se_option = DittoSEOption.DITTO_SE_OPTION_RS
        else:
            raise NotImplementedError(
                "se_option {0} is not valid SE option for Laplacian computation".format(se_option)
            )
        return compute_laplacian_se(
            est_result=self.__final_est_result._ptr, model=self._model_ptr, se_option=ditto_se_option
        )

    def __do_compute_se_saem(
        self,
        se_option: typing.Literal["S", "R", "RS"] | typing.Literal["LINEAR", "SA"] | None = None,
        saem_se_settings: DittoSASESetting | None = None,
    ) -> npt.NDArray[np.float_]:
        if saem_se_settings is None:
            saem_se_settings = DittoSASESetting(
                n_mode1=2,
                n_mode2=0,
                n_mode3=2,
                n_mode4=0,
                kappa_min=0.01,
                kappa_max=100,
                negative_feedback=0.7,
                optimal_acceptance=0.3,
                alpha=1.0,
                n_iter=50,
                n_chain=2,
                n_burnin=5
            )

        ditto_saem_se_option: DittoSAEMSEOption

        if se_option == "LINEAR":
            ditto_saem_se_option = DittoSAEMSEOption.DITTO_SAEM_SE_OPTION_LIN
        elif se_option == "SA":
            ditto_saem_se_option = DittoSAEMSEOption.DITTO_SAEM_SE_OPTION_SA
        elif se_option is None:
            ditto_saem_se_option = DittoSAEMSEOption.DITTO_SAEM_SE_OPTION_SA
        else:
            raise NotImplementedError("se_option {0} is not valid SE option for SAEM SE computation".format(se_option))

        return compute_saem_se(
            est_result=self.__final_est_result._ptr,
            model=self._model_ptr,
            se_option=ditto_saem_se_option,
            se_settings=saem_se_settings
        )

        """

        Args:
            title: The title of the individual fit plot, which defaults to "Individual Fit Plot"
            idv_columns: The name of the idv column used by plotting, if you do not define the column defined by the defined idv column when the model was built
            dv_columns:The name of the dv column used by plotting, if you do not define the column defined by the defined dv column when the model was built

            label_ipred: The text label of the legend of the individual prediction which defaults to "IPRED"
            rows: The number of rows of the subgraph when plotting, if not defined, it will be automatically calculated
            cols: The number of columns of the subgraph when plotting, if not defined,  it will be automatically calculated

        Returns:

        if (not isinstance(dv_columns, str | None)):
            raise TypeError("'dv_columns' should be type str.")
        if (not isinstance(pred_columns, str)):
            raise TypeError("'pred_columns' should be type str.")
        if (not isinstance(ipred_columns, str)):
            raise TypeError("'ipred_columns' should be type str.")
        if (not isinstance(label_dv, str)):
            raise TypeError("'label_dv' should be type str.")
        if (not isinstance(label_pred, str)):
            raise TypeError("'label_pred' should be type str.")
        if (not isinstance(label_ipred, str)):
            raise TypeError("'label_ipred' should be type str.")
        if (not isinstance(rows, int | None)):
            raise TypeError("'rows' should be type int.")
        if (not isinstance(cols, int | None)):
            raise TypeError("'cols' should be type int.")

        idv_name: str
        dv_name: str
        subject_name: str | None
        reserved_colnames = self.__data.reserved_colnames
        if (idv_columns is None):
            if (reserved_colnames.get("idv") is None):
                raise ValueError("Neither the 'idv' is defined when modeling or plotting the individual fits.")
            else:
                idv_name = str(reserved_colnames["idv"])
        else:
            idv_name = idv_columns
        if (dv_columns is None):
            if (reserved_colnames.get("dv") is None):
                raise ValueError("Neither the 'dv' is defined when modeling or plotting the individual fits.")
            else:
                dv_name = str(reserved_colnames["dv"])
        else:
            dv_name = dv_columns
        if (reserved_colnames.get("subject") is None):
            subject_name = None
        else:
            subject_name = str(reserved_colnames["subject"])

        data: pd.DataFrame = self.pred
        ind_fit_plotly: IndFitPlot = IndFitPlot(
            data=data,
            title=title,
            subject_id_name=subject_name,
            meta=dict(),
            idv_columns=idv_name,
            dv_columns=dv_name,
            pred_columns=pred_columns,
            ipred_columns=ipred_columns,
            label_dv=label_dv,
            label_pred=label_pred,
            label_ipred=label_ipred,
            sub_rows=rows,
            sub_cols=cols,
            is_plot_subgraph=True
        )
        return ind_fit_plotly

[docs]    def plot_iteration_trace(self) -> IterPlot:
        name_param: list[str] = ["iteration", "Objective Function Value"]
        for i in self._mod.thetas:
            name_param.append(i.name)
        for i in self._mod.etas:
            name_param.append(i.name)
        for i in self._mod.epsilons:
            name_param.append(i.name)
        iteration_trace: pd.DataFrame = pd.DataFrame(columns=name_param)

        iteration_count: int = 1
        for iteration in self.iter_details:
            iteration_i: dict[str, list[float]] = {}
            iteration_i["iteration"] = [iteration_count]
            iteration_i["Objective Function Value"] = [iteration.obj_val]
            param_index: int = 2
            for theta_index in range(len(iteration.pop_params.theta)):
                iteration_i[name_param[param_index]] = [iteration.pop_params.theta[theta_index]]
                param_index += 1
            for omega_index in range(len(iteration.pop_params.omega)):
                iteration_i[name_param[param_index]] = [iteration.pop_params.omega[omega_index][omega_index]]
                param_index += 1
            for sigma_index in range(len(iteration.pop_params.sigma)):
                iteration_i[name_param[param_index]] = [iteration.pop_params.sigma[sigma_index][sigma_index]]
                param_index += 1
            if len(iteration_trace.columns) == len(iteration_i):
                iteration_trace = pd.concat([iteration_trace, pd.DataFrame(iteration_i)], axis=0)
            else:
                raise ValueError("An error occurred in the iteration history generation")
            iteration_count += 1

        iter_plotly: IterPlot = IterPlot(iteration_trace)
        return iter_plotly
