[docs]    def insert_param_deriv_inline(
        self,
        node: ast.FunctionDef,
        order_of_deriv: typing.Literal[0, 1, 2],
        dAdt_deriv: bool = True,
        simplify: bool = True
    ) -> dict[str, list[str]]:
        """通过 ast 遍历 node 的语法树, 为 parsing_term_names 中的所有变量添加对 var_context 中的变量的偏导

        Args:
            node (ast.FunctionDef): node 语法树

        Returns:
            dict[str, list[str]]: 变量名为 key, 其对应的所有偏导的变量名为 value 的字典
        """
        result: dict[str, list[str]] = {}
        appending_assignments: list[ast.Assign | ast.AugAssign] = []

        if self._mod_typ == "ode":
            if dAdt_deriv:
                if order_of_deriv >= 1:

                    dAdt_wrt_1st_mp: dict[AutoDiffDADtWrt, sympy.Expr] = {}
                    for dAdt_wrt_key, dAdt_wrt_val in self._dAdt_wrt_mp.items():
                        if dAdt_wrt_key.wrt2nd is None:
                            dAdt_wrt_1st_mp[dAdt_wrt_key] = dAdt_wrt_val

                    exprs = [*dAdt_wrt_1st_mp.values()]
                    if simplify:
                        refined_expressions, assignments, cse_ctx = self._cse(exprs)
                        appending_assignments.extend(assignments)
                    else:
                        refined_expressions = exprs
                        cse_ctx = None

                    dAdt_wrt_1st_keys = [*dAdt_wrt_1st_mp.keys()]

                    if len(dAdt_wrt_1st_keys) != len(exprs):
                        raise ValueError("Length mismatched")

                    while dAdt_wrt_1st_keys:
                        key = dAdt_wrt_1st_keys.pop(0)  # pop first
                        expr = refined_expressions.pop(0)
                        cmt = key.cmt
                        wrt = key.wrt
                        translated_deriv = self._sympy_translator.translate(expr, captured_ctx=cse_ctx)
                        appending_assignments.append(
                            ast.Assign(
                                targets=(_make_ode_subscript_ast(_DADT_MP, (cmt, wrt)),),
                                value=translated_deriv,
                                lineno=None
                            )
                        )

                if order_of_deriv >= 2:
                    dAdt_wrt_2nd_mp: dict[AutoDiffDADtWrt, sympy.Expr] = {}
                    for dAdt_wrt_key, dAdt_wrt_val in self._dAdt_wrt_mp.items():
                        if dAdt_wrt_key.wrt2nd:
                            dAdt_wrt_2nd_mp[dAdt_wrt_key] = dAdt_wrt_val

                    exprs = [*dAdt_wrt_2nd_mp.values()]
                    if simplify:
                        refined_expressions, assignments, cse_ctx = self._cse(exprs)
                        appending_assignments.extend(assignments)
                    else:
                        refined_expressions = exprs
                        cse_ctx = None

                    dAdt_wrt_2nd_keys = [*dAdt_wrt_2nd_mp.keys()]

                    if len(dAdt_wrt_2nd_keys) != len(exprs):
                        raise ValueError("Length mismatched")

                    while dAdt_wrt_2nd_keys:
                        key = dAdt_wrt_2nd_keys.pop(0)  # pop first
                        expr = refined_expressions.pop(0)
                        cmt = key.cmt
                        wrt = key.wrt
                        wrt2nd = key.wrt2nd
                        if not wrt2nd:
                            raise ValueError("Invalid wrt2nd")
                        translated_deriv = self._sympy_translator.translate(expr, captured_ctx=cse_ctx)
                        appending_assignments.append(
                            ast.Assign(
                                targets=(_make_ode_subscript_ast(_DADT_MP, (cmt, wrt, wrt2nd)),),
                                value=translated_deriv,
                                lineno=None
                            )
                        )

        if order_of_deriv >= 1:
            target_wrt_1st_mp: dict[AutoDiffTargetWrt, sympy.Expr] = {}
            for target_wrt_key, target_wrt_val in self._target_wrt_mp.items():
                if target_wrt_key.wrt2nd is None:
                    target_wrt_1st_mp[target_wrt_key] = target_wrt_val

            exprs = [*target_wrt_1st_mp.values()]
            captured_ctx = self._local_context.copy()
            if simplify:
                refined_expressions, assignments, cse_ctx = self._cse(exprs)
                appending_assignments.extend(assignments)
                captured_ctx = captured_ctx + cse_ctx
            else:
                refined_expressions = exprs
                captured_ctx = None

            target_wrt_1st_keys = [*target_wrt_1st_mp.keys()]
            if len(target_wrt_1st_keys) != len(exprs):
                raise ValueError("Length mismatched")

            while target_wrt_1st_keys:
                key = target_wrt_1st_keys.pop(0)  # pop first
                expr = refined_expressions.pop(0)
                target_name = key.target_name
                wrt_var = key.wrt
                masked_key = mask_variable(target_name)
                masked_term_name = f"{masked_key}_wrt_{wrt_var.name}"
                if masked_key not in result.keys():
                    result[masked_key] = [masked_term_name]
                else:
                    result[masked_key].append(masked_term_name)

                ast_expr = self._sympy_translator.translate(expr, captured_ctx=captured_ctx)
                appending_assignments.append(
                    ast.Assign(targets=(ast.Name(id=masked_term_name),), value=ast_expr, lineno=None)
                )

        if order_of_deriv >= 2:
            target_wrt_2nd_mp: dict[AutoDiffTargetWrt, sympy.Expr] = {}
            for target_wrt_key, target_wrt_val in self._target_wrt_mp.items():
                if target_wrt_key.wrt2nd is not None:
                    target_wrt_2nd_mp[target_wrt_key] = target_wrt_val

            exprs = [*target_wrt_2nd_mp.values()]
            captured_ctx = self._local_context.copy()
            if simplify:
                refined_expressions, assignments, cse_ctx = self._cse(exprs)
                appending_assignments.extend(assignments)
                captured_ctx = captured_ctx + cse_ctx
            else:
                refined_expressions = exprs
                captured_ctx = None

            target_wrt_2nd_keys = [*target_wrt_2nd_mp.keys()]
            if len(target_wrt_2nd_keys) != len(exprs):
                raise ValueError("Length mismatched")

            while target_wrt_2nd_keys:
                key = target_wrt_2nd_keys.pop(0)  # pop first
                expr = refined_expressions.pop(0)
                target_name = key.target_name
                wrt_var = key.wrt
                wrt2nd_var = key.wrt2nd
                if wrt2nd_var is None:
                    raise KeyError
                masked_key = mask_variable(target_name)
                masked_term_name = f"{masked_key}_wrt_{wrt_var.name}_wrt_{wrt2nd_var.name}"
                if masked_key not in result.keys():
                    result[masked_key] = [masked_term_name]
                else:
                    result[masked_key].append(masked_term_name)

                ast_expr = self._sympy_translator.translate(expr, captured_ctx=captured_ctx)
                appending_assignments.append(
                    ast.Assign(targets=(ast.Name(id=masked_term_name),), value=ast_expr, lineno=None)
                )

        node.body.extend(appending_assignments)

        return result



[docs]    def insert_dosing_deriv_inline(self, node: ast.FunctionDef) -> None:
        appending_assignments: list[ast.stmt] = []

        if self._mod_typ == "ode":  # only handle this with ode type
            for alag_expr in self._alag_mp.values():
                for var in self._var_context.values():
                    wrt_v = var

                    if not isinstance(var, Eta):  # 只需要对 eta 的偏导
                        continue

                    partial_deriv = SYMPY_ZERO
                    if isinstance(alag_expr, sympy.Expr):
                        partial_deriv = alag_expr.diff(wrt_v)

                    for cmt in self._cmt_context.values():
                        if partial_deriv != 0:
                            # NOTE: 修改 A[dosing_cmt, alag] 的初值，用于计算 DADalag
                            # NONMEM 做法，数学原理见 TODO: issue#xxx
                            translated_deriv = self._sympy_translator.translate(partial_deriv * -1)
                            assignment = ast.Assign(
                                targets=(_make_ode_subscript_ast(_A_MP, (cmt, wrt_v)),),
                                value=ast.BinOp(
                                    left=_make_ode_subscript_ast(_DADT_MP, cmt), op=ast.Mult(), right=translated_deriv
                                ),
                                lineno=None
                            )
                            appending_assignments.append(assignment)

            if self._rate_mp and not self._dur_mp:
                for rate_expr in self._rate_mp.values():
                    for var in self._var_context.values():
                        wrt_v = var

                        if not isinstance(var, Eta):  # 只需要对 eta 的偏导
                            continue

                        partial_deriv = SYMPY_ZERO
                        if isinstance(rate_expr, sympy.Expr):
                            partial_deriv = rate_expr.diff(wrt_v)
                        cmt_index = 0
                        for cmt in self._cmt_context.values():
                            cmt_index += 1
                            if partial_deriv != 0:
                                appending_assignments.extend(
                                    self.__get_dose_rate_deriv(cmt, wrt_v, partial_deriv, cmt_index)
                                )

            elif self._dur_mp and not self._rate_mp:
                for dur_expr in self._dur_mp.values():
                    for var in self._var_context.values():
                        wrt_v = var

                        if not isinstance(var, Eta):  # 只需要对 eta 的偏导
                            continue

                        partial_deriv = SYMPY_ZERO
                        if isinstance(dur_expr, sympy.Expr):
                            partial_deriv = dur_expr.diff(wrt_v)
                        cmt_index = 0
                        for cmt in self._cmt_context.values():
                            cmt_index += 1
                            if partial_deriv != 0:
                                # 添加 dur 导数
                                appending_assignments.extend(
                                    self.__get_dose_dur_deriv(cmt, wrt_v, partial_deriv, cmt_index)
                                )

            elif self._dur_mp and self._rate_mp:
                raise ValueError("Both rate and duration are not supported as model parameters")

            if self._f_mp:
                for f_expr in self._f_mp.values():
                    for var in self._var_context.values():
                        wrt_v = var

                        if not isinstance(var, Eta):  # 只需要对 eta 的偏导
                            continue

                        partial_deriv = SYMPY_ZERO
                        if isinstance(f_expr, sympy.Expr):
                            partial_deriv = f_expr.diff(wrt_v)
                        cmt_index = 0
                        for cmt in self._cmt_context.values():
                            cmt_index += 1
                            if partial_deriv != 0:
                                # 添加 f 导数
                                appending_assignments.extend(
                                    self.__get_dose_f_deriv(cmt, wrt_v, partial_deriv, cmt_index)
                                )

        node.body.extend(appending_assignments)

        appending_assignments: list[ast.Assign] = []

        if self._mod_typ == "ode":  # only handle this with ode type

                                    value=ast.Attribute(attr=cmt.name, value=ast.Name('self'), lineno=None),

                                )
                                if a0_cmt.name == cmt.name:
                                    assignment = ast.Assign(
                                        targets=(_make_ode_subscript_ast(_A_MP, (cmt, wrt_v)),),
                                        value=ast.BinOp(
                                            left=f_expr,
                                            op=ast.Mult(),
                                            right=self._sympy_translator.translate(partial_deriv),
                                            lineno=None
                                        ),
                                        lineno=None
                                    )
                                else:
                                    assignment = ast.Assign(
                                        targets=(_make_ode_subscript_ast(_A_MP, (cmt, wrt_v)),),
                                        value=ast.Constant(0),
                                        lineno=None
                                    )
                                appending_assignments.append(assignment)

        node.body.extend(appending_assignments)


    def __get_dose_dur_deriv(self, cmt: Compartment, wrt_v: SymVar, partial_deriv: typing.Any,
                                cmt_index: int) -> list[ast.stmt]:
        appending_assignments: list[ast.stmt] = []
        # 添加 dur 导数
        f_expr = ast.Attribute(
            attr="fraction", value=ast.Attribute(attr=cmt.name, value=ast.Name(id='self'), lineno=None), lineno=None
        )
        rate_expr = _make_dose_params_attribute_ast(_DOSE_PARAMS, "rate")

        rate_div_dur = ast.BinOp(
            left=rate_expr,
            op=ast.Div(),
            right=ast.Attribute(
                attr="duration",
                value=ast.Attribute(attr=cmt.name, value=ast.Name(id='self'), lineno=None),
                lineno=None
            )
        )
        translated_deriv = self._sympy_translator.translate(partial_deriv)  # dur/d_eta
        _dduration = ast.BinOp(left=f_expr, op=ast.Mult(), right=rate_div_dur, lineno=None)
        dduration = ast.BinOp(left=_dduration, op=ast.Mult(), right=translated_deriv, lineno=None)
        # 滴注开始
        assignment = ast.Assign(
            targets=(_make_ode_subscript_ast(_DADT_MP, (cmt, wrt_v)),),
            value=ast.BinOp(
                left=_make_ode_subscript_ast(_DADT_MP, (cmt, wrt_v)),
                op=ast.Add(),
                right=ast.UnaryOp(op=ast.USub(), operand=dduration),
                lineno=None
            ),
            lineno=None
        )
        appending_assignments.append(
            ast.If(
                test=ast.Call(
                    func=_make_dose_params_attribute_ast(_DOSE_PARAMS, "is_infusion_d_start"),
                    args=[ast.Constant(cmt_index - 1)],
                    kwargs=None,
                    keywords=[]
                ),
                body=[assignment],
                orelse=[],
                lineno=None
            )
        )

        # 滴注结束
        dadt_assignment = ast.Assign(
            targets=(_make_ode_subscript_ast(_DADT_MP, (cmt, wrt_v)),),
            value=ast.BinOp(
                left=_make_ode_subscript_ast(_DADT_MP, (cmt, wrt_v)),
                op=ast.Sub(),
                right=ast.UnaryOp(op=ast.USub(), operand=dduration),
                lineno=None
            ),
            lineno=None
        )
        body = []
        body.append(dadt_assignment)

        a_assignment = ast.Assign(
            targets=(_make_ode_subscript_ast(_A_MP, (cmt, wrt_v)),),
            value=ast.BinOp(
                left=_make_ode_subscript_ast(_A_MP, (cmt, wrt_v)),
                op=ast.Add(),
                right=ast.BinOp(left=rate_expr, op=ast.Mult(), right=translated_deriv, lineno=None),
                lineno=None
            ),
            lineno=None
        )
        body.append(a_assignment)
        test = ast.Call(
            func=_make_dose_params_attribute_ast(_DOSE_PARAMS, "is_infusion_d_end"),
            args=[ast.Constant(cmt_index - 1)],
            kwargs=None,
            keywords=[]
        )

        appending_assignments.append(ast.If(test=test, body=body, orelse=[], lineno=None))
        return appending_assignments

    def __get_dose_rate_deriv(self, cmt: Compartment, wrt_v: SymVar, partial_deriv: typing.Any,
                                cmt_index: int) -> list[ast.stmt]:
        appending_assignments: list[ast.stmt] = []
        # 添加 rate 导数

        drate_deta = self._sympy_translator.translate(partial_deriv)  # drate/d_eta

        # 滴注开始
        assignment = ast.Assign(
            targets=(_make_ode_subscript_ast(_DADT_MP, (cmt, wrt_v)),),
            value=ast.BinOp(left=_make_ode_subscript_ast(_DADT_MP, (cmt, wrt_v)), op=ast.Add(), right=drate_deta),
            lineno=None
        )
        appending_assignments.append(
            ast.If(
                test=ast.Call(
                    func=_make_dose_params_attribute_ast(_DOSE_PARAMS, "is_infusion_r_start"),
                    args=[ast.Constant(cmt_index - 1)],
                    kwargs=None,
                    keywords=[]
                ),
                body=[assignment],
                orelse=[],
                lineno=None
            )
        )

        # 滴注结束
        body = []
        dadt_assignment = ast.Assign(
            targets=(_make_ode_subscript_ast(_DADT_MP, (cmt, wrt_v)),),
            value=ast.BinOp(left=_make_ode_subscript_ast(_DADT_MP, (cmt, wrt_v)), op=ast.Sub(), right=drate_deta),
            lineno=None
        )
        body.append(dadt_assignment)
        dur_expr = _make_dose_params_attribute_ast(_DOSE_PARAMS, "dur")
        a_assignment = ast.Assign(
            targets=(_make_ode_subscript_ast(_A_MP, (cmt, wrt_v)),),
            value=ast.BinOp(
                left=_make_ode_subscript_ast(_A_MP, (cmt, wrt_v)),
                op=ast.Sub(),
                right=ast.BinOp(left=dur_expr, op=ast.Mult(), right=drate_deta, lineno=None),
                lineno=None
            ),
            lineno=None
        )
        body.append(a_assignment)

        appending_assignments.append(
            ast.If(
                test=ast.Call(
                    func=_make_dose_params_attribute_ast(_DOSE_PARAMS, "is_infusion_r_end"),
                    args=[ast.Constant(cmt_index - 1)],
                    kwargs=None,
                    keywords=[]
                ),
                body=body,
                orelse=[],
                lineno=None
            )
        )
        return appending_assignments

    def __get_dose_f_deriv(self, cmt: Compartment, wrt_v: SymVar, partial_deriv: typing.Any,
                            cmt_index: int) -> list[ast.stmt]:
        appending_assignments: list[ast.stmt] = []

        df_deta = self._sympy_translator.translate(partial_deriv)  # df/d_eta
        dadt_deta = ast.BinOp(
            left=_make_dose_params_attribute_ast(_DOSE_PARAMS, "amt"), op=ast.Mult(), right=df_deta, lineno=None
        )

        # bolus
        assignment = ast.Assign(
            targets=(_make_ode_subscript_ast(_A_MP, (cmt, wrt_v)),),
            value=ast.BinOp(left=_make_ode_subscript_ast(_A_MP, (cmt, wrt_v)), op=ast.Add(), right=dadt_deta),
            lineno=None
        )
        appending_assignments.append(
            ast.If(
                test=ast.Call(
                    func=_make_dose_params_attribute_ast(_DOSE_PARAMS, "is_bolus"),
                    args=[ast.Constant(cmt_index - 1)],
                    kwargs=None,
                    keywords=[]
                ),
                body=[assignment],
                orelse=[],
                lineno=None
            )
        )

        # 滴注结束
        a_assignment = ast.Assign(
            targets=(_make_ode_subscript_ast(_A_MP, (cmt, wrt_v)),),
            value=ast.BinOp(
                left=_make_ode_subscript_ast(_A_MP, (cmt, wrt_v)),
                op=ast.Add(),
                right=ast.BinOp(
                    left=_make_dose_params_attribute_ast(_DOSE_PARAMS, "amt"),
                    op=ast.Mult(),
                    right=ast.Attribute(
                        attr="fraction",
                        value=ast.Attribute(attr=cmt.name, value=ast.Name('self'), lineno=None),
                        lineno=None
                    ),
                    lineno=None
                ),
                lineno=None
            ),
            lineno=None
        )

        appending_assignments.append(
            ast.If(

[docs]    def visit_If(self, node: ast.If) -> typing.Any:
        transformer = IfElseTransformer(
            source_code=self._source_code, global_context=self._global_context, local_context=self._local_context
        )
        transformed_node: typing.Any = transformer.visit(node)

        _node: list[ast.stmt]
        if isinstance(transformed_node, typing.Iterable):
            _node = typing.cast(list[ast.stmt], transformed_node)
        else:
            _node = [transformed_node]
        transformed: list[ast.stmt] = []

        for child_node in _node:
            if isinstance(child_node, ast.Assign):
                if child_node not in transformer.variable_hoist_transformer.hoisting_assignments.values():
                    transformed.append(self.visit_Assign(child_node))
                else:
                    transformed.append(child_node)
            else:
                transformed.append(child_node)

        return transformed

[docs]    def visit_Assign(self, node: ast.Assign) -> typing.Any:
        lhs = node.targets
        _node: typing.Any = node

        # deal with diff
        if len(lhs) == 1:
            token = lhs[0]

            maybe_assign_subroutine = self.__subroutine is None
            node.value = rhs = self.visit(node.value)

            if self.__subroutine is not None and maybe_assign_subroutine:
                # subroutine is newly assigned
                subroutine_ast, subroutine_obj = self.__subroutine
                self._local_context[_SolutionOutMp.name] = _SolutionOutMp(subroutine=subroutine_obj)
                _node = [
                    *self._create_subroutine_args(subroutine=subroutine_obj),
                    ast.Assign(targets=(ast.Name(id=_SolutionOutMp.name),), value=subroutine_ast, lineno=None),
                    node
                ]

            if isinstance(token, ast.Name):
                if token.id not in self._local_context.keys():
                    rhs_expr = eval(ast.unparse(rhs), self._global_context.as_dict(), self._local_context.as_dict())
                else:
                    rhs_expr = getattr(self._local_context, token.id)
                # rhs_expr = eval(ast.unparse(node.value), self._global_context.as_dict(), self._local_context.as_dict())
                self._local_context[token.id] = rhs_expr

                if isinstance(rhs_expr, PKAnalyticalSolution):
                    node.value = ast.Name(id=_SolutionOutMp.name)

                if token.id in self.target_names:  # 只对返回值 y 求导
                    self._target_wrt_mp.update(self._get_target_wrt(target=token.id, value=rhs_expr))

            elif isinstance(token, ast.Attribute):
                _token = self.visit(token)
                lhs_expr = eval(ast.unparse(_token), self._global_context.as_dict(), self._local_context.as_dict())
                _node.targets[0] = _token

                if isinstance(lhs_expr, DADt):
                    rhs_expr = eval(ast.unparse(rhs), self._global_context.as_dict(), self._local_context.as_dict())
                    self._dAdt_wrt_mp.update(self._get_dAdt_wrt(target=lhs_expr, value=rhs_expr))  # dAdt 对 eta 的一阶导
                elif isinstance(lhs_expr, Alag):
                    rhs_expr = eval(ast.unparse(rhs), self._global_context.as_dict(), self._local_context.as_dict())
                    dosing_cmt = lhs_expr.cmt
                    if dosing_cmt not in self._alag_mp.keys():
                        self._alag_mp[dosing_cmt] = rhs_expr
                elif isinstance(lhs_expr, Rate):
                    rhs_expr = eval(ast.unparse(rhs), self._global_context.as_dict(), self._local_context.as_dict())
                    dosing_cmt = lhs_expr.cmt

                    if dosing_cmt not in self._rate_mp.keys():
                        self._rate_mp[dosing_cmt] = rhs_expr
                elif isinstance(lhs_expr, Duration):
                    rhs_expr = eval(ast.unparse(rhs), self._global_context.as_dict(), self._local_context.as_dict())
                    dosing_cmt = lhs_expr.cmt

                    if dosing_cmt not in self._dur_mp.keys():
                        self._dur_mp[dosing_cmt] = rhs_expr
                elif isinstance(lhs_expr, CmtInitialA):
                    rhs_expr = eval(ast.unparse(rhs), self._global_context.as_dict(), self._local_context.as_dict())
                    dosing_cmt = lhs_expr.cmt

                    if dosing_cmt not in self._a0_mp.keys():
                        self._a0_mp[dosing_cmt] = rhs_expr
                elif isinstance(lhs_expr, Fraction):
                    # TODO: @xuchongyi 补充 Rate | Duration | Fraction | A0 的 梯度
                    rhs_expr = eval(ast.unparse(rhs), self._global_context.as_dict(), self._local_context.as_dict())
                    dosing_cmt = lhs_expr.cmt

                    if dosing_cmt not in self._f_mp.keys():
                        self._f_mp[dosing_cmt] = rhs_expr
                else:
                    raise NotImplementedError()

            else:
                evaluated = eval(ast.unparse(token), self._global_context.as_dict(), self._local_context.as_dict())
                if isinstance(evaluated, DADtWrt | DADt):
                    pass
                else:
                    raise NotImplementedError(evaluated)
        else:
            raise NotImplementedError()

        return _node


    def _get_target_wrt(self, target: str, value: typing.Any) -> dict[AutoDiffTargetWrt, sympy.Expr]:
        # partial_deriv_assignments: list[ast.Assign] = []
        target_wrt: dict[AutoDiffTargetWrt, sympy.Expr] = {}
        for wrt_index, wrt in enumerate(self._var_context.values()):
            if not isinstance(wrt, Eta | Eps):  # NOTE: dydp 只需要对 eta, eps 的一阶导
                continue

            key1st = AutoDiffTargetWrt(target_name=target, wrt=wrt)

            if isinstance(value, sympy.Expr):
                partial_deriv = value.diff(wrt)
            else:  # 任何常数对任何参数求导都是 0
                partial_deriv = SYMPY_ZERO

            if self._mod_typ == "ode":
                for cmt in self._cmt_context.values():
                    if isinstance(value, sympy.Expr) and isinstance(wrt, Eta):
                        A_wrt_var = AWrt(cmt=cmt, wrt=wrt)
                        partial_deriv += value.diff(cmt.A) * A_wrt_var

            elif self._mod_typ == "advan":
                if self.__subroutine is None:
                    raise ValueError("Subroutine trans is not defined")
                if isinstance(value, sympy.Expr) and isinstance(wrt, Eta):
                    solved_F_wrt_var = PKSolvedFWrt(subroutine=self.__subroutine[1], wrt=wrt)
                    partial_deriv += value.diff(PKSolvedF(subroutine=self.__subroutine[1])) * solved_F_wrt_var

                    for cmt_index in range(self._n_cmt):
                        solved_A_wrt_var = PKSolvedAWrt(subroutine=self.__subroutine[1], index=cmt_index, wrt=wrt)
                        partial_deriv += value.diff(
                            PKSolvedA(subroutine=self.__subroutine[1], index=cmt_index)
                        ) * solved_A_wrt_var

            target_wrt[key1st] = target_wrt.get(key1st, SYMPY_ZERO) + partial_deriv

            for wrt2nd_index, wrt2nd in enumerate(self._var_context.values()):
                # NOTE: dydp 只需要对 eta & eps 的二阶混合偏导 (Y)
                if self._pred_variant == "y":
                    if not (isinstance(wrt, Eta) and isinstance(wrt2nd, Eps)):
                        continue
                # NOTE: dydp 只需要对 eta & eta 的二阶混合偏导 (LIKELIHOOD)
                else:
                    if not (isinstance(wrt, Eta) and isinstance(wrt2nd, Eta) and wrt_index <= wrt2nd_index):
                        continue

                key2nd = AutoDiffTargetWrt(target_name=target, wrt=wrt, wrt2nd=wrt2nd)

                if self._mod_typ == "ode" and self._pred_variant != "y":
                    if isinstance(value, sympy.Expr):
                        partial_deriv_2nd = value.diff(wrt, wrt2nd)
                    else:  # 任何常数对任何参数求导都是 0
                        partial_deriv_2nd = SYMPY_ZERO

                    # special case, 2nd for eta is required in ode form if is LIKELIHOOD Modeling
                    if isinstance(wrt, Eta) and isinstance(wrt2nd, Eta):
                        for cmt1_index, cmt1 in enumerate(self._cmt_context.values()):
                            if isinstance(value, sympy.Expr):
                                A1_wrt_var1 = AWrt(cmt=cmt1, wrt=wrt)
                                partial_deriv_2nd += value.diff(cmt1.A, wrt2nd) * A1_wrt_var1

                                A1_wrt_var2 = AWrt(cmt=cmt1, wrt=wrt2nd)
                                partial_deriv_2nd += value.diff(cmt1.A, wrt) * A1_wrt_var2

                                A1_wrt_var1_var2 = AWrt(cmt=cmt1, wrt=wrt, wrt2nd=wrt2nd)
                                partial_deriv_2nd += value.diff(cmt1.A) * A1_wrt_var1_var2

                                for cmt2_index, cmt2 in enumerate(self._cmt_context.values()):
                                    if cmt1_index <= cmt2_index:
                                        A2_wrt_var1 = AWrt(cmt=cmt2, wrt=wrt)
                                        # A2_wrt_var2 = AWrt(cmt=cmt2, wrt=wrt2nd)

                                        # partial_deriv_2nd += value.diff(cmt1.A, cmt2.A) * A1_wrt_var1 * A2_wrt_var2
                                        partial_deriv_2nd += value.diff(cmt2.A, cmt1.A) * A2_wrt_var1 * A1_wrt_var2
                else:
                    if isinstance(value, sympy.Expr):
                        partial_deriv_2nd = partial_deriv.diff(wrt2nd)
                    else:  # 任何常数对任何参数求导都是 0
                        partial_deriv_2nd = SYMPY_ZERO

                # TODO: FIXME: 这里需要 ADVAN / deta 的二阶导
                # elif self._mod_typ == "advan":
                #     if isinstance(value, sympy.Expr):
                #         solved_F_wrt_var = PKSolvedFWrt(wrt=wrt)
                #         partial_deriv += value.diff(_PK_SOLVED_F) * solved_F_wrt_var

                target_wrt[key2nd] = target_wrt.get(key2nd, SYMPY_ZERO) + partial_deriv_2nd

        return target_wrt

    def _get_dAdt_wrt(self, target: DADt, value: typing.Any) -> dict[AutoDiffDADtWrt, sympy.Expr]:
        dAdt_wrt: dict[AutoDiffDADtWrt, sympy.Expr] = {}
        if self._mod_typ != "ode":
            return dAdt_wrt
        for wrt1_index, wrt1 in enumerate(self._var_context.values()):
            if not isinstance(wrt1, Eta):  # NOTE: dAdt/dp 只需要对 eta 的一阶导
                continue

            key1 = AutoDiffDADtWrt(cmt=target.cmt, wrt=wrt1)

            partial_deriv: sympy.Expr
            if isinstance(value, sympy.Expr):
                partial_deriv = value.diff(wrt1)
            else:  # 任何常数对任何参数求导都是 0
                partial_deriv = SYMPY_ZERO

            for cmt in self._cmt_context.values():
                if isinstance(value, sympy.Expr):
                    A_wrt_var1 = AWrt(cmt=cmt, wrt=wrt1)
                    partial_deriv += value.diff(cmt.A) * A_wrt_var1

            dAdt_wrt[key1] = dAdt_wrt.get(key1, SYMPY_ZERO) + partial_deriv

            # if it is ode and we are predicting likelihood, we need 2nd order derivative
            if self._mod_typ == "ode" and self._pred_variant != "y":
                for wrt2_index, wrt2 in enumerate(self._var_context.values()):
                    # only case we need to compute 2nd order ode
                    if not (isinstance(wrt1, Eta) and isinstance(wrt2, Eta) and wrt1_index <= wrt2_index):
                        continue

                    key2 = AutoDiffDADtWrt(cmt=target.cmt, wrt=wrt1, wrt2nd=wrt2)

                    partial_deriv_2nd: sympy.Expr
                    if isinstance(value, sympy.Expr):
                        partial_deriv_2nd = value.diff(wrt1, wrt2)
                    else:  # 任何常数对任何参数求导都是 0
                        partial_deriv_2nd = SYMPY_ZERO

                    for cmt1_index, cmt1 in enumerate(self._cmt_context.values()):
                        if isinstance(value, sympy.Expr):
                            dAdt_dp1_dA1 = value.diff(wrt1, cmt1.A)
                            A1_wrt_p2 = AWrt(cmt=cmt1, wrt=wrt2)
                            partial_deriv_2nd += dAdt_dp1_dA1 * A1_wrt_p2

                            dAdt_dA1_dp1 = value.diff(cmt1.A, wrt1)
                            partial_deriv_2nd += dAdt_dA1_dp1 * A1_wrt_p2

                            for cmt2_index, cmt2 in enumerate(self._cmt_context.values()):
                                if cmt1_index <= cmt2_index:
                                    dAdt_dA1_dA2 = value.diff(cmt1.A, cmt2.A)
                                    A1_wrt_p1 = AWrt(cmt=cmt1, wrt=wrt1)
                                    A2_wrt_p2 = AWrt(cmt=cmt2, wrt=wrt2)

                                    partial_deriv_2nd += dAdt_dA1_dA2 * A1_wrt_p1 * A2_wrt_p2

                            A1_wrt_p1_p2 = AWrt(cmt=cmt1, wrt=wrt1, wrt2nd=wrt2)
                            dAdt_dA1 = value.diff(cmt1.A)
                            partial_deriv_2nd += dAdt_dA1 * A1_wrt_p1_p2

                    dAdt_wrt[key2] = dAdt_wrt.get(key2, SYMPY_ZERO) + partial_deriv_2nd

        # for cmt1_index, cmt1 in enumerate(self._cmt_context.values()):
        #     wrt = cmt1.A
        #     key1 = AutoDiffDADtWrt(cmt=target.cmt, wrt=wrt.cmt)

        #         partial_deriv = SYMPY_ZERO

        #     dAdt_wrt[key1] = dAdt_wrt.get(key1, SYMPY_ZERO) + partial_deriv

    def _create_subroutine_args(self, subroutine: PKAnalyticalSolution[typing.Any]) -> list[ast.Assign]:
        assignments: list[ast.Assign] = []
        for param_name, param_expr in subroutine.params.items():
            if param_expr:
                if isinstance(param_expr, sympy.Expr):
                    translated_param = self._sympy_translator.translate(param_expr)
                    assignments.append(
                        ast.Assign(
                            targets=(_make_solution_in_subscript_ast(param_name=param_name),),
                            value=translated_param,
                            lineno=None
                        )
                    )

                elif isinstance(param_expr, int | float):
                    assignments.append(
                        ast.Assign(
                            targets=(_make_solution_in_subscript_ast(param_name),), value=param_expr, lineno=None
                        )
                    )

                for wrt in self._var_context.values():
                    if not isinstance(wrt, Eta):  # NOTE: dAdt/dp 只需要对 eta 的一阶导
                        continue

                    if isinstance(param_expr, sympy.Expr):
                        partial_deriv = param_expr.diff(wrt)
                    else:  # 任何常数对任何参数求导都是 0
                        partial_deriv = SYMPY_ZERO

                    translated_param = self._sympy_translator.translate(partial_deriv)
                    assignments.append(
                        ast.Assign(
                            targets=(_make_solution_in_subscript_ast(param_name=param_name, wrt=wrt),),
                            value=translated_param,
                            lineno=None
                        )
                    )

        # assignments.append(ast.Assign(targets=(ast.Name(id=solution_in.name), )))
        self._local_context[_SolutionInMp.name] = _SolutionInMp(subroutine=subroutine)
        return assignments

    def _cse(self, exprs: list[sympy.Expr]) -> tuple[list[sympy.Expr], list[ast.Assign], VarContext[sympy.Expr]]:
        cse_ctx = VarContext[sympy.Expr]()
        captured_ctx = self._local_context.copy()  # 这个 context 专门用来处理临时变量
        appending_assignments: list[ast.Assign] = []
        common_sub_expressions, refined_expressions = sympy.cse(exprs, symbols=sympy.numbered_symbols(prefix="__"))

        sub_expr_term: sympy.Symbol
        sub_expr: sympy.Expr
        ast_expr: ast.expr
        for sub_expr_term, sub_expr in common_sub_expressions:
            cse_ctx[sub_expr_term.name] = sub_expr
            ast_expr = self._sympy_translator.translate(sub_expr, captured_ctx=captured_ctx + cse_ctx)

            appending_assignments.append(
                ast.Assign(targets=(ast.Name(id=sub_expr_term.name),), value=ast_expr, lineno=None)
            )

        return refined_expressions, appending_assignments, cse_ctx
