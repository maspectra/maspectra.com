class Eps(SymVar):
    """
    Residual Variance
    """
    __slots__ = ("_sigma", "_index")

    def __new__(cls, name: str, init_value: ValueType | None = None, fixed: bool = False, **kwargs: typing.Any) -> Eps:
        return typing.cast(Eps, super().__new__(cls, name, init_value, fixed, **kwargs))

    @property
    def sigma(self) -> Sigma:
        """
        Sigma Matrix for epsilon
        """
        return self._sigma

    @sigma.setter
    def sigma(self, sigma: Sigma) -> None:
        """
        Set Sigma Matrix for epsilon

        Args:
            sigma: Sigma Matrix
        """

        if not isinstance(sigma, Sigma):
            raise ValueError(f'sigma must be an instance of Sigma, not {type(sigma)}')
        self._sigma = sigma

    def _set_index(self, index: int) -> None:
        self._index = index

    def _get_index(self) -> int:
        return self._index


class Sigma:
    """
    Residual Variance Matrix
    """

    def __init__(self, names: list[str], init_values: npt.NDArray[np.float_], fixed: bool) -> None:
        self.__names = names
        self.init_values = init_values
        self.fixed = fixed

    @classmethod
    def from_ndarray(cls, array: npt.NDArray[np.float_], fixed: bool) -> Sigma:
        names: list[str] = [_eta.name for _eta in np.diag(array)]
        init_values: npt.NDArray[np.float_] = np.full(array.shape, np.nan, np.float_)

        for row in range(array.shape[0]):
            for col in range(array.shape[1]):
                cell = array[row][col]
                if isinstance(cell, Eps):
                    if row != col:
                        raise ValueError("Eps can only appear on diagonal")
                    init_values[row][col] = cell.init_value

                else:
                    init_values[row][col] = cell

        return cls(names=names, init_values=init_values, fixed=fixed)

    @property
    def names(self) -> list[str]:
        return self.__names

[docs]def sigma(
    init_sigma: ValueType | Block | npt.NDArray[np.float_] | list[list[ValueType]] | None = None,
    fixed: bool = False
) -> Eps | list[Eps]:
    sigma_mat: npt.NDArray[np.float_]

    if init_sigma is None:
        # 默认 Sigma 构建，对角线为 1 的 1x1 Sigma 矩阵
        sigma_mat = np.array([[1]], dtype=float)
    elif isinstance(init_sigma, int | float):
        # 单一值的 Sigma 构建，将 eps 初值放置于 1x1 Sigma 的对角线即可
        sigma_mat = np.array([[init_sigma]], dtype=float)

    elif isinstance(init_sigma, Block):
        # 使用下三角构建 Sigma 矩阵
        sigma_mat = init_sigma.as_full()

    elif isinstance(init_sigma, typing.Iterable):
        # init_sigma 是一个 ListLike 的对象 (List / npt.NDArray[np.float_])
        arr = np.array(init_sigma, dtype=float)
        if len(arr.shape) != 2:
            raise ValueError("Only 2-d Sigma matrix is allowed, not a {0} dimensional array".format(len(arr.shape)))

        if arr.shape[0] != arr.shape[1]:
            raise ValueError(
                "Can only construct Sigma with block matrix, not a shape of {0} x {1}".format(
                    arr.shape[0], arr.shape[1]
                )
            )

        sigma_mat = arr

    else:
        raise TypeError("Invalid type of init_sigma {0}".format(type(init_sigma)))

    n_dim = sigma_mat.shape[0]

    # 校验是否是对称矩阵
    for i in range(1, n_dim):
        if not np.all(np.diag(sigma_mat, -i) == np.diag(sigma_mat, i)):
            raise ValueError("Given Sigma \n {0} \n is not symmetric".format(str(sigma_mat)))

    _names: list[str] = []

    for i in range(n_dim):
        _names.append(f"__unnamed_eps_{generate_nanoid(6)}")

    epsilons: list[Eps] = []
    sigma_diag = np.diag(sigma_mat)
    for index, init_value in enumerate(sigma_diag):
        eps = Eps(name=_names[index], init_value=init_value, fixed=fixed)
        eps._set_index(index)
        epsilons.append(eps)

    sigma = Sigma.from_ndarray(
        array=sigma_mat - np.diag(np.diag(sigma_mat)) + np.diag(np.array(epsilons, dtype=object)), fixed=fixed
    )

    for _eta in epsilons:
        _eta.sigma = sigma

    return epsilons if len(epsilons) != 1 else epsilons[0]
