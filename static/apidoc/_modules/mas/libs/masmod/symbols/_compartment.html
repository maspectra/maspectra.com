def _make_ode_subscript_ast(
    dummy_mp: _AMp | _DADtMp,
    key: Compartment | tuple[Compartment, SymVar | Compartment] |
    tuple[Compartment, SymVar | Compartment, SymVar | Compartment]
) -> ast.Subscript:
    value = ast.Name(id=dummy_mp.name)
    if isinstance(key, tuple):
        elts: list[ast.Attribute] = []
        for part in key:
            elts.append(ast.Attribute(value=ast.Name(id="self"), attr=part.name))
        slice = ast.Tuple(dims=elts, elts=elts)
        return ast.Subscript(slice=slice, value=value)
    else:
        return ast.Subscript(slice=ast.Attribute(value=ast.Name(id="self"), attr=key.name), value=value)


class _DADtMp:  # Dummy
    name: str = "__dAdt"

    def __setitem__(self, key: typing.Any, value: typing.Any) -> None:
        raise NotImplementedError()

    @typing.overload
    def __getitem__(self, key: Compartment) -> DADt:
        ...

    @typing.overload
    def __getitem__(self, key: tuple[Compartment, SymVar] | tuple[Compartment, SymVar, SymVar]) -> DADtWrt:
        ...

    def __getitem__(
        self, key: Compartment | tuple[Compartment, SymVar] | tuple[Compartment, SymVar, SymVar]
    ) -> DADt | DADtWrt:
        if isinstance(key, tuple):
            if len(key) == 2:  # first order
                cmt, wrt = key
                return DADtWrt(cmt=cmt, wrt=wrt)
            else:  # second order
                cmt, wrt, wrt2nd = key
                return DADtWrt(cmt=cmt, wrt=wrt, wrt2nd=wrt2nd)
        else:
            return key.dAdt


_DADT_MP = _DADtMp()

class DADt(sympy.Symbol):
    __slots__ = ("_cmt", "_expr")

    def __new__(cls, cmt: Compartment) -> DADt:
        name = f"dA_{cmt.name}dt"
        instance = typing.cast(DADt, super().__new__(cls, name))
        instance._cmt = cmt
        instance._expr = sympy.Number(0.)
        return instance

    @property
    def cmt(self) -> Compartment:
        return self._cmt

    @property
    def expr(self) -> Expression:
        return self._expr

class Compartment:

    def __init__(
        self,
        name: str,
        init_value: float | None = None,
        default_dose: bool = False,
        default_obs: bool = False,
        alag: float | None = None,
        fraction: float | None = None,
        rate: float | None = None,
        duration: float | None = None
    ) -> None:
        self.__name = name
        self.__init_value = CmtInitialA(cmt=self)
        if init_value:
            self.__init_value._expr = init_value

        self.__default_dose = default_dose
        self.__default_obs = default_obs

        self.__dAdt = DADt(cmt=self)

        self.__alag = Alag(cmt=self)
        if alag is not None:
            self.__alag._expr = alag

        self.__fraction = Fraction(cmt=self)
        if fraction is not None:
            self.__fraction._expr = fraction

        self.__rate = Rate(cmt=self)
        if rate is not None:
            self.__rate._expr = rate

        self.__duration = Duration(cmt=self)
        if duration is not None:
            self.__duration._expr = duration

        self.__A = CmtSolvedA(cmt=self)

    def __deepcopy__(self, memo: typing.Any) -> Compartment:
        return Compartment(
            name=self.__name,
            init_value=self.__init_value._expr,
            default_dose=self.default_dose,
            default_obs=self.default_obs,
            alag=self.alag._expr,
            fraction=self.__fraction._expr,
            rate=self.__rate._expr,
            duration=self.__duration._expr
        )

    @property
    def A(self) -> CmtSolvedA:
        """Amount of the compartment solved from differential equation

        Returns:
            CmtSolvedA: Amount of the compartment
        """
        return self.__A

    @property
    def dAdt(self) -> DADt:
        """Differential equation of the compartment

        Returns:
            DADt: the expression formula of differential equation

        Examples:
            >>> central_cmt.dAdt = ka * dose_cmt.A - k * central_cmt.A
        """
        return self.__dAdt

    @dAdt.setter
    def dAdt(self, val: Expression) -> None:
        if not isinstance(val, int | float | sympy.Expr):
            raise TypeError()

        self.__dAdt._expr = val

    @property
    def name(self) -> str:
        """Name of the compartment

        Returns:
            str: name
        """
        return self.__name

    @name.setter
    def name(self, val: str) -> None:
        self.__name = val
        self.__A.name = f"A_{val}"
        self.__dAdt.name = f"dA_{val}dt"
        self.__alag.name = f"Alag_{val}"
        self.__fraction.name = f"Fraction_{val}"
        self.__rate.name = f"Rate_{val}"
        self.__duration.name = f"Duration_{val}"

    @property
    def alag(self) -> Alag:
        """Absorption Lag of the compartment

        Returns:
            Alag: the expression formula of alag
        """
        return self.__alag

    @alag.setter
    def alag(self, val: Expression) -> None:
        if not isinstance(val, int | float | sympy.Expr):
            raise TypeError()
        self.__alag._expr = val

    @property
    def fraction(self) -> Fraction:
        """BioAvailability of the compartment

        Returns:
            Fraction: the expression formula of BioAvailability
        """
        return self.__fraction

    @fraction.setter
    def fraction(self, val: Expression) -> None:
        if not isinstance(val, int | float | sympy.Expr):
            raise TypeError()
        self.__fraction._expr = val

    @property
    def rate(self) -> Rate:
        """Dosing rate of the compartment

        Returns:
            Rate: the expression formula of dosing rate
        """
        return self.__rate

    @rate.setter
    def rate(self, val: Expression) -> None:
        if not isinstance(val, int | float | sympy.Expr):
            raise TypeError()
        self.__rate._expr = val

    @property
    def duration(self) -> Duration:
        """Dosing duration of the compartment

        Returns:
            Duration: the expression formula of dosing duration
        """
        return self.__duration

    @duration.setter
    def duration(self, val: Expression) -> None:
        if not isinstance(val, int | float | sympy.Expr):
            raise TypeError()
        self.__duration._expr = val

    @property
    def init_value(self) -> CmtInitialA:
        """Initial amount of the compartment A_0

        Returns:
            CmtInitialA: the expression formula of initial amount
        """
        return self.__init_value

    @init_value.setter
    def init_value(self, val: Expression) -> None:
        if not isinstance(val, int | float | sympy.Expr):

        self.__init_value._expr = val

    def default_dose(self) -> bool:
        """If the compartment is set to be default dosing compartment

        Returns:
            bool: True if default dosing

        return self.__default_dose

        return self.__default_obs

    Returns:
        Compartment: compartment variable
    """
    if init_value is None:
        init_value = 0.
