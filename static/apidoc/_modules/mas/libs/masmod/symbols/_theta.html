from __future__ import annotations

import typing

from mas.libs.masmod.symbols._common import DEFAULT_LOWER_BOUND, DEFAULT_UPPER_BOUND, BoundsType, ValueType
from mas.libs.masmod.symbols._symvar import SymVar
from mas.libs.phanpy.uid.nanoid import generate_nanoid

__all__ = ["theta", "Theta"]


class Theta(SymVar):
    """
    固定效应的变量
    """
    __slots__ = ("_bounds", "_mu_ref")

    def __new__(
        cls,
        name: str,
        init_value: ValueType | None = None,
        bounds: BoundsType | None = None,
        fixed: bool = False,
        **kwargs: typing.Any
    ) -> Theta:
        instance = typing.cast(Theta, super().__new__(cls, name, init_value, fixed, **kwargs))
        instance.bounds = bounds
        return instance

    @property
    def bounds(self) -> tuple[float, float]:
        """Boundary of the Theta variable"""
        return self._bounds

    @bounds.setter
    def bounds(self, bounds: BoundsType | None) -> None:
        if bounds is not None:
            _lower, _upper = bounds

            if _lower is not None and type(_lower) not in [float, int]:
                raise TypeError("指定的 lower_bound 数据类型错误，必须是 float 或者 int, 传入了 {0}".format(type(_lower)))
            elif _lower is None:
                _lower = DEFAULT_LOWER_BOUND
            else:
                _lower = float(_lower)

            if _upper is not None and not type(_upper) in [float, int]:
                raise TypeError("指定的 upper_bound 数据类型错误，必须是 float 或者 int, 传入了 {0}".format(type(_upper)))
            elif _upper is None:
                _upper = DEFAULT_UPPER_BOUND
            else:
                _upper = float(_upper)

            bounds = (_lower, _upper)

        else:
            bounds = (DEFAULT_LOWER_BOUND, DEFAULT_UPPER_BOUND)

        self._bounds = bounds

    @property
    def mu_ref(self) -> bool:
        # default value
        if not hasattr(self, "_mu_ref"):
            return False
        return self._mu_ref

    @mu_ref.setter
    def mu_ref(self, mu_ref: bool) -> None:
        if hasattr(self, "_mu_ref") and mu_ref != self.mu_ref:
            raise ValueError("无法重复设定 {0} 的 mu, 是不是在 if/else 中定义了 mu?".format(self.name))
        self._mu_ref = mu_ref
