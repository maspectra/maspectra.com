class ValueType(enum.Enum):
    VALUE_TYPE_VOID = "void"

    VALUE_TYPE_DOUBLE = "double"
    VALUE_TYPE_INT = "int"
    VALUE_TYPE_BOOL = "bool"
    VALUE_TYPE_LONG = "long"
    VALUE_TYPE_STRING = "std::string"

    VALUE_TYPE_VEC = "Eigen::VectorXd"
    VALUE_TYPE_VEC_REF = "Eigen::VectorXd&"
    VALUE_TYPE_VEC_CONST_REF = "const Eigen::VectorXd&"
    VALUE_TYPE_MAT = "Eigen::MatrixXd"
    VALUE_TYPE_MAT_REF = "Eigen::MatrixXd&"
    VALUE_TYPE_MAT_CONST_REF = "const Eigen::MatrixXd&"
    VALUE_TYPE_CUBE_REF = "CubeXd&"

    VALUE_TYPE_CMT_PTR = "Compartment*"
    VALUE_TYPE_CMT_SYS_REF = "CompartmentSystem&"

    VALUE_TYPE_DOUBLE_CONTEXT = "VarContext<double>"
    VALUE_TYPE_DOUBLE_CONTEXT_REF = "VarContext<double>&"
    VALUE_TYPE_STRING_CONTEXT = "VarContext<std::string>"
    VALUE_TYPE_STRING_CONTEXT_REF = "VarContext<std::string>&"

    # Special Case for Subroutine
    VALUE_TYPE_SUBROUTINE = "Subroutine"
    VALUE_TYPE_SUBROUTINE_REF = "Subroutine&"
    VALUE_TYPE_PK_PARAM = "PkParameters"
    VALUE_TYPE_PK_PARAM_CONST_REF = "const PkParameters&"
    VALUE_TYPE_DOSING_PARAM = "DosingParameters"
    VALUE_TYPE_DOSING_PARAM_CONST_REF = "const DosingParameters&"

    def to_cc_type(self) -> str:
        return self.value

    def is_numeric(self) -> bool:
        return self.value == ValueType.VALUE_TYPE_DOUBLE.value or \
            self.value == ValueType.VALUE_TYPE_INT.value or \
                self.value == ValueType.VALUE_TYPE_LONG.value

    def can_be_eq_to(self, other: ValueType) -> bool:
        """与 A == B 不同, 这里会判断 Subroutine 和 numeric 的逻辑"""
        if (self.value == ValueType.VALUE_TYPE_SUBROUTINE.value and other.value == ValueType.VALUE_TYPE_SUBROUTINE_REF.value) or \
            (self.value == ValueType.VALUE_TYPE_SUBROUTINE_REF.value and other.value == ValueType.VALUE_TYPE_SUBROUTINE.value):
            return True

        if self.is_numeric() and other.is_numeric():
            return True

        return self == other

    @classmethod
    def from_val(cls, val: typing.Any) -> ValueType:
        typ: ValueType | None = None

        if isinstance(val, float):
            typ = ValueType.VALUE_TYPE_DOUBLE
        elif isinstance(val, bool):
            typ = ValueType.VALUE_TYPE_INT
        elif isinstance(val, int):
            if val <= -2147483648 or val >= 2147483647:
                typ = ValueType.VALUE_TYPE_LONG
            else:
                typ = ValueType.VALUE_TYPE_INT
        elif isinstance(val, str):
            typ = ValueType.VALUE_TYPE_STRING

        if typ is None:
            raise TypeError("不支持的 const 类型: {0}".format(type(val)))

        return typ

    @classmethod
    def from_dtype(cls, dtype: typing.Any) -> ValueType:
        typ: ValueType | None = None

        if pandas.api.types.is_int64_dtype(dtype):
            typ = ValueType.VALUE_TYPE_INT
        elif pandas.api.types.is_float_dtype(dtype):
            typ = ValueType.VALUE_TYPE_DOUBLE
        if typ is None:
            raise TypeError("不支持的 dtype: {0}".format(dtype))
        return typ

[docs]    def translate(self) -> list[str]:
        """主入口，翻译 python ast => cc syntax"""
        if self._module_type == "ode":
            module_type = "ModuleType::MODULE_TYPE_ODE"
        elif self._module_type == "advan":
            module_type = "ModuleType::MODULE_TYPE_SUBROUTINE"
            if self._subroutine is None:
                raise TypeError("Using ADVAN but subroutine is not defined")
        else:
            module_type = "ModuleType::MODULE_TYPE_PRED"

        if self._pred_variant == "y":
            pred_variant = "PredVariant::PRED_VARIANT_Y"
        elif self._pred_variant == "likelihood":
            pred_variant = "PredVariant::PRED_VARIANT_LIKELIHOOD"
        else:
            pred_variant = "PredVariant::PRED_VARIANT_M2LIKELIHOOD"

        translated: list[str] = [
            *self.__include_headers(),
            "",  # empty line
            "class __Module : public IModule",
            "{",
            "public:",
            "ModuleType module_type() const",
            "{",
            "  return {};".format(module_type),
            "}",
            "PredVariant pred_variant() const",
            "{",
            "  return {};".format(pred_variant),
            "}",
        ]

        for part in self._cls_def.body:
            if isinstance(part, ast.FunctionDef):
                if part.name in self._trans_functions.keys():
                    translated.extend(self._do_translate_func(part, self._trans_functions[part.name]))
            elif isinstance(part, ast.Expr):
                value = part.value
                if isinstance(value, ast.Constant) and isinstance(value.value, str):
                    pass
                else:
                    self.__raise(part, NotImplementedError("class Expr {0} is not implemented yet".format(value)))
            else:
                self.__raise(part, NotImplementedError("class 功能 {0} is not implemented yet".format(part)))

        translated.append("};")

        translated.extend(self.__module_factory_function())

        return translated


    def _do_translate_func(self, func_def: ast.FunctionDef, func_signature: FuncSignature) -> list[str]:
        """翻译函数体"""
        translated: list[str] = []

        ctx: Ctx = {}

        # 处理函数签名
        signature: FuncSignature
        if func_def.name in self._trans_functions.keys():
            signature = self._inject_self_to_signature(self._trans_functions[func_def.name])
        else:
            signature = self._extract_func_signature(func_def)

        self._assert_func_signature_is_valid(func_def, signature)

        for arg_name, arg_type in signature.args.items():
            ctx[arg_name] = arg_type

        for var_name in self._var_context.keys():
            ctx[mask_self_attr(var_name)] = ValueType.VALUE_TYPE_DOUBLE

        if self._module_type == "ode":
            ctx[INTEGRAL_T.name] = ValueType.VALUE_TYPE_DOUBLE
            for cmt_name in self._cmt_context.keys():
                ctx[mask_self_attr(cmt_name)] = ValueType.VALUE_TYPE_CMT_PTR

        elif self._module_type == "advan":
            ctx[_SolutionInMp.name] = ValueType.VALUE_TYPE_PK_PARAM

        for data_indexer_name, data_indexer_obj in self._dat_context.items():
            value_type: ValueType
            if data_indexer_obj.typ == "continuous":
                value_type = ValueType.VALUE_TYPE_DOUBLE
            elif data_indexer_obj.typ == "categorical":
                value_type = ValueType.VALUE_TYPE_STRING
            else:
                raise TypeError()
            ctx[mask_self_attr(data_indexer_name)] = value_type

        signature = self._inject_result_container_to_signature(signature)

        translated.extend(["", *self._do_translate_func_signature(func_def, signature), "{"])

        translated_body: list[str] = [
            "", "// #region 从 self context 获取变量值", *self._retrieve_vars_from_self(), "// #endregion", ""
        ]

        # 处理函数内容
        for stmt in func_def.body:
            translated_body.extend(self._do_translate_statement(stmt, ctx))

        translated.append("// 类型申明")
        for var_name, var_type in ctx.items():
            if var_name not in signature.args.keys():
                if var_type == ValueType.VALUE_TYPE_PK_PARAM:
                    translated.append(f"{var_type.value} {var_name}({len(self._etas)});")
                else:
                    translated.append(f"{var_type.value} {var_name};")
        translated.extend([*translated_body])

        translated.extend(
            [
                "",
                "// #region 将 reserved self attr 赋值至 __container",
                f"__container.resize({len(func_signature.result_variables)});"
            ]
        )

        for var_index, result_var_name in enumerate(func_signature.result_variables):
            if result_var_name not in ctx.keys():
                raise ValueError(f"没有输出 {result_var_name}")
            translated.append(f"__container({var_index}) = {result_var_name};")

        translated.extend(["// #endregion"])

        if func_signature.update_context:
            translated.append("// #region 导出临时变量")
            for var_name, var_type in ctx.items():
                if var_name.startswith("__"):
                    # protected variable , pass assign
                    continue

                if var_name in signature.args.items():
                    # 入参, 不导出
                    continue

                if var_type.is_numeric():
                    translated.append(f"{DOUBLE_CONTEXT_VARNAME}[\"{var_name}\"] = {var_name};")
                elif var_type == ValueType.VALUE_TYPE_STRING:
                    translated.append(f"{STRING_CONTEXT_VARNAME}[\"{var_name}\"] = {var_name};")
                else:
                    continue  # 无法处理的类型, 放弃导出

            translated.extend(["// #endregion"])
        translated.append("return Ok();")
        translated.extend(["}"])

        return translated

    def _do_translate_statement(self, statement: ast.stmt, ctx: Ctx) -> list[str]:
        """翻译 statement"""
        translated: list[str] = []
        if self.__locatable(statement):
            translated.append(f"// {self._source_code_lines[statement.lineno - 1].strip()}")
        if isinstance(statement, ast.Assign):
            translated.extend(self._do_translate_assign(statement, ctx))
        elif isinstance(statement, ast.If):
            translated.extend(self._do_translate_if(statement, ctx))
        elif isinstance(statement, ast.Expr):
            translated.extend(self._do_translate_expr_stmt(statement, ctx))
        elif isinstance(statement, ast.Return):
            # special case
            # TODO: 这个会影响用户自定义 closure 函数，但是暂时不支持
            pass
        elif isinstance(statement, ast.AugAssign):
            translated.extend(self._do_translate_assign(statement, ctx))
        elif isinstance(statement, ast.Pass):
            # you pass, i pass
            pass
        else:
            self.__raise(
                statement, NotImplementedError("statement {0} is not implemented yet".format(type(statement)))
            )

        return translated

    def _do_translate_assign(self, assign: ast.Assign | ast.AugAssign, ctx: Ctx) -> list[str]:
        """翻译赋值 block"""
        translated: list[str] = []

        if isinstance(assign, ast.Assign):
            targets = assign.targets

            if len(targets) != 1:
                self.__raise(assign, ValueError("Assigning to more then one target is not implemented yet"))

            target = self._do_eval_expr(targets[0], ctx)
        else:
            target = self._do_eval_expr(assign.target, ctx)

        value = self._do_eval_expr(assign.value, ctx)

        if value.typ is None:
            self.__raise(assign, ValueError("Variable must have a certain type"))

        if target.is_name:
            if target.v in ctx.keys():
                _declared_typ = ctx[target.v]
                # 不允许重新定义类型
                # 特殊情况: 两个变量都是数值类型，那么使用 double 处理
                if not _declared_typ.can_be_eq_to(value.typ):
                    self.__raise(
                        assign,
                        TypeError(
                            "Change type of variable {0} from {1} to {2} is forbidden".format(
                                target.v, _declared_typ, value.typ
                            )
                        )
                    )
                ctx[target.v] = value.typ
            else:
                ctx[target.v] = value.typ

        op: str
        if isinstance(assign, ast.Assign):
            op = "="
        else:
            op = f"{self._do_eval_op(assign.op)}="

        translated.append(f"{target.v} {op} {value.v};")
        return translated

    def _do_translate_if(self, if_: ast.If, ctx: Ctx, is_else_if: bool = False) -> list[str]:
        """翻译 if block"""
        translated: list[str] = []

        # if 的比较条件
        test_condition = self._do_eval_expr(if_.test, ctx)

        if test_condition.typ is None or not test_condition.typ.is_numeric():
            self.__raise(
                if_, TypeError("The condition of if statement must be bool not {0}".format(test_condition.typ))
            )

        if self.__locatable(if_):
            translated.append(f"// {self._source_code_lines[if_.lineno - 1].strip()}")

        branch_pref = "if" if not is_else_if else "else if "
        translated.extend([f"{branch_pref} ({test_condition.v})", "{"])

        # body
        for stmt in if_.body:
            translated.extend(self._do_translate_statement(stmt, ctx))

        translated.append("}")

        # 如果是 else if
        if len(if_.orelse) == 1 and isinstance(if_.orelse[0], ast.If):
            translated.extend(self._do_translate_if(if_.orelse[0], ctx, is_else_if=True))
        else:  # 如果是 else
            if if_.orelse:
                translated.append("else {")
                for stmt in if_.orelse:
                    translated.extend(self._do_translate_statement(stmt, ctx))

                translated.append("}")

        return translated

    def _do_translate_expr_stmt(self, expr_stmt: ast.Expr, ctx: Ctx) -> list[str]:
        translated: list[str] = []

        stmt_token = expr_stmt.value

        if isinstance(stmt_token, ast.Call):
            func_ = eval(ast.unparse(stmt_token.func), self._global_ctx, {})

            if id(func_) == id(print):
                translated.append(
                    "std::cout << {} << std::endl;".format(
                        " << \" \" << ".join([ast.unparse(token).replace("'", "\"") for token in stmt_token.args])
                    )
                )

        return translated

    def _do_eval_expr(self, expr: ast.expr, ctx: Ctx) -> EvaluatedExpr:
        """处理一个表达式"""
        # 如果是个表达式，直接 eval
        if isinstance(expr, ast.Expression):
            return self._do_eval_expr(expr.body, ctx)

        # 如果是 Name
        if isinstance(expr, ast.Name):
            _identifier = expr.id
            _typ: ValueType | None = None
            # 如果在上下文有对应的类型提供，返回其类型，否则是 None
            if _identifier in ctx.keys():
                _typ = ctx[_identifier]

            return EvaluatedExpr(v=_identifier, is_name=True, typ=_typ, token=expr)

        # 如果是一个常数
        if isinstance(expr, ast.Constant):
            return self._do_eval_constant(expr)

        # 如果是一个比较 e.g. a > b
        if isinstance(expr, ast.Compare):
            if len(expr.ops) != 1:
                self.__raise(expr, ValueError("Multiple compare operators are not supported"))

            if len(expr.comparators) != 1:
                self.__raise(expr, ValueError("Multiple comparators are not supported"))
            left = self._do_eval_expr(expr.left, ctx)
            ops = self._do_eval_cmpop(expr.ops[0])
            right = self._do_eval_expr(expr.comparators[0], ctx)

            # TODO: add type validation for left and right

            if (not left.typ or not right.typ):
                self.__raise(expr, ValueError("Cannot infer correct type"))

            # 只能比较两个数值类型或者比较符号两边的类型相同
            if not (left.typ.is_numeric() and right.typ.is_numeric()) and left.typ != right.typ:
                self.__raise(expr, TypeError("Types for two expressions do not match"))

            return EvaluatedExpr(v=f"{left.v} {ops.v} {right.v}", token=expr, typ=ValueType.VALUE_TYPE_INT)

        # 如果是属性访问
        if isinstance(expr, ast.Attribute):
            # 访问的对象指针
            value = self._do_eval_expr(expr.value, ctx)
            attr = expr.attr
            # 检查是否对 self 有非法访问
            if value.v == "self":
                if attr not in self._self_ctx.keys():
                    self.__raise(expr, AttributeError("Accessing self {0} is forbidden".format(attr)))

                # 获取 self 属性的类型
                obj = self._self_ctx[attr]
                typ: ValueType
                if isinstance(obj, SymVar) or isinstance(obj, ColumnIndexer):
                    typ = ctx[mask_self_attr(attr)]
                elif isinstance(obj, Compartment):
                    typ = ValueType.VALUE_TYPE_CMT_PTR
                elif issubclass(obj, PKAnalyticalSolution):
                    typ = ValueType.VALUE_TYPE_SUBROUTINE
                    return EvaluatedExpr(v=typ.name, typ=typ, token=expr)
                else:
                    try:
                        evaluated_const = self._do_eval_constant(ast.Constant(value=obj))
                        if evaluated_const.typ is None:
                            raise ValueError("Not-Reachable")
                        typ = evaluated_const.typ
                    except Exception as e:
                        self.__raise(expr, e)
                # 由于会对 self 中的所有字段执行重新赋值，这里直接返回 attr
                return EvaluatedExpr(v=mask_self_attr(attr), typ=typ, token=expr)
            elif ctx.get(value.v) == ValueType.VALUE_TYPE_CMT_PTR:
                if attr not in [
                    "alag", "fraction", "duration", "rate", "init_value"
                ]:  # TODO: @xuchongyi FIX hardcode attribute match
                    self.__raise(expr, AttributeError("{0} 没有 {1} 属性".format(unmask_self_attr(value.v), attr)))
                return EvaluatedExpr(v=f"{value.v}->{attr}", typ=ValueType.VALUE_TYPE_DOUBLE, token=expr)
            elif value.v == _DOSE_PARAMS.name:
                if attr in _DOSE_PARAMS._property:
                    return EvaluatedExpr(v=f"{value.v}.{attr}", token=expr, typ=ValueType.VALUE_TYPE_DOUBLE)
                elif attr in _DOSE_PARAMS._function:
                    return EvaluatedExpr(
                        v=f"{value.v}.{attr}", token=expr, typ=ValueType.VALUE_TYPE_DOSING_PARAM_CONST_REF
                    )
                else:
                    self.__raise(expr, AttributeError("{0} 没有 {1} 属性".format(unmask_self_attr(value.v), attr)))

            elif value.v in self._global_ctx.keys():
                return EvaluatedExpr(v=f"{value.v}.{attr}", token=expr)
            else:
                self.__raise(expr, NotImplementedError("Object usage other than 'self' is not implemented yet"))

        # 如果是函数调用
        if isinstance(expr, ast.Call):
            # 获取函数名称
            func = self._do_eval_expr(expr.func, ctx)

            if func.typ == ValueType.VALUE_TYPE_SUBROUTINE:
                if not self._subroutine:
                    self.__raise(expr, ValueError("Instantiating Subroutine without defining subroutine"))
                passthrough_args = filter(
                    lambda ctx_key: ctx_key.startswith("__") and ctx_key.endswith("_"), [*ctx.keys()]
                )

                return EvaluatedExpr(
                    v=
                    f"*(_try_({_SolutionOutMp.name}.solve(Advan::ADVAN{self._subroutine.advan}, Trans::TRANS{self._subroutine.trans}, {_SolutionInMp.name}, {', '.join(passthrough_args)})))",
                    typ=ValueType.VALUE_TYPE_SUBROUTINE,
                    token=expr
                )
            if func.typ == ValueType.VALUE_TYPE_DOSING_PARAM_CONST_REF:
                if len(expr.args) == 1 and isinstance(expr.args[0], ast.Constant):
                    return EvaluatedExpr(v=f"{func.v}({expr.args[0].value})", typ=ValueType.VALUE_TYPE_INT, token=expr)
                else:
                    self.__raise(
                        expr, ValueError("Cannot handle function {0} with args nums {1}".format(func, len(expr.args)))
                    )
            # 获取函数参数
            args = ", ".join(map(lambda arg: self._do_eval_expr(arg, ctx).v, expr.args))

            func_obj = eval(func.v, self._global_ctx, {})
            if id(func_obj) == id(mu):
                return EvaluatedExpr(v=args, typ=ValueType.VALUE_TYPE_DOUBLE, token=expr)
            elif id(func_obj) in _functor_mapper.keys():
                return EvaluatedExpr(
                    v=f"{_functor_mapper[id(func_obj)]}({args})", typ=ValueType.VALUE_TYPE_DOUBLE, token=expr
                )
            else:
                self.__raise(expr, ValueError("Cannot handle function {0}".format(func)))

        # 如果是二元运算
        if isinstance(expr, ast.BinOp):
            # 左表达式
            left = self._do_eval_expr(expr.left, ctx)

            # 右表达式
            right = self._do_eval_expr(expr.right, ctx)

            # TODO: 特殊处理 mu ref
            # if left.typ == ValueType.VALUE_TYPE_MU_DOUBLE

            # 二元运算要求左右都是数值类型，二元运算的返回值只会是 double 类型
            # special case: bool * 数值类型在乘法运算中可以被允许
            computable: bool = left.typ is not None and right.typ is not None and \
                (left.typ.is_numeric() and right.typ.is_numeric())
            if not computable:
                self.__raise(expr, ValueError("Binary Ops can be applied to numerical types only"))

            # 特殊处理 pow `a ** b`
            if isinstance(expr.op, ast.Pow):
                return EvaluatedExpr(v=f"pow({left.v}, {right.v})", typ=ValueType.VALUE_TYPE_DOUBLE, token=expr)
            else:
                op = self._do_eval_op(expr.op)
                return EvaluatedExpr(v=f"({left.v}{op.v}{right.v})", typ=ValueType.VALUE_TYPE_DOUBLE, token=expr)

        # 如果是一元运算，即 -a, +b
        if isinstance(expr, ast.UnaryOp):
            op = self._do_eval_unaryop(typing.cast(ast.unaryop, expr.op))
            oprand = self._do_eval_expr(expr.operand, ctx)
            # 根据 oprand 的具体类型
            return EvaluatedExpr(v=f"({op.v}{oprand.v})", typ=oprand.typ, token=expr)

        # 如果是
        if isinstance(expr, ast.BoolOp):
            op = self._do_eval_boolop(typing.cast(ast.boolop, expr.op))

            fragments: list[str] = []
            for val in expr.values:
                fragments.append(f"({self._do_eval_expr(val, ctx).v})")

            return EvaluatedExpr(v=op.v.join(fragments), typ=ValueType.VALUE_TYPE_INT, token=expr)

        if isinstance(expr, ast.Subscript):
            local_ctx: dict[str, typing.Any] = {
                "self": self._self_ctx
            }
            if self._module_type == "advan" and self._subroutine:
                local_ctx[_SolutionInMp.name] = _SolutionInMp(subroutine=self._subroutine)
                local_ctx[_SolutionOutMp.name] = _SolutionOutMp(subroutine=self._subroutine)

            for ctx_key, ctx_v_typ in ctx.items():
                # Special case to handle reassigning subroutine output
                if ctx_v_typ.can_be_eq_to(ValueType.VALUE_TYPE_SUBROUTINE) and ctx_key not in local_ctx.keys():
                    local_ctx[ctx_key] = local_ctx[_SolutionOutMp.name]
            evaluated = eval(ast.unparse(expr), self._global_ctx, local_ctx)
            index: int
            name = ast.unparse(expr.value)
            if isinstance(evaluated, DADt | CmtSolvedA):
                index = self._compute_ode_index(evaluated.cmt)
            elif isinstance(evaluated, DADtWrt | AWrt):
                if evaluated.wrt2nd:
                    index = self._compute_ode_index(evaluated.cmt, evaluated.wrt, evaluated.wrt2nd)
                else:
                    index = self._compute_ode_index(evaluated.cmt, evaluated.wrt)
            elif isinstance(evaluated, PKParamArg):
                param_name, tuple_indexer = self._compute_pk_param_key(evaluated.param_name)
                # special case, pk argument
                return EvaluatedExpr(
                    v=f"{name}[\"{param_name}\"]({tuple_indexer[0]}, {tuple_indexer[1]})",
                    typ=ValueType.VALUE_TYPE_DOUBLE,
                    token=expr
                )
            elif isinstance(evaluated, PKParamArgWrt):
                param_name, tuple_indexer = self._compute_pk_param_key(evaluated.param_name, wrt=evaluated.wrt)
                # special case, pk argument wrt
                return EvaluatedExpr(
                    v=f"{name}[\"{param_name}\"]({tuple_indexer[0]}, {tuple_indexer[1]})",
                    typ=ValueType.VALUE_TYPE_DOUBLE,
                    token=expr
                )
            elif isinstance(evaluated, PKSolvedF):
                return EvaluatedExpr(v=f"{name}(0, 0)", typ=ValueType.VALUE_TYPE_DOUBLE, token=expr)
            elif isinstance(evaluated, PKSolvedA):
                row_index, col_index = self._compute_pk_solution_index(evaluated.index, None)
                return EvaluatedExpr(
                    v=f"{name}({row_index}, {col_index})", typ=ValueType.VALUE_TYPE_DOUBLE, token=expr
                )
            elif isinstance(evaluated, PKSolvedFWrt):
                row_index, col_index = self._compute_pk_solution_index(None, evaluated.wrt)
                return EvaluatedExpr(
                    v=f"{name}({row_index}, {col_index})", typ=ValueType.VALUE_TYPE_DOUBLE, token=expr
                )
            elif isinstance(evaluated, PKSolvedAWrt):
                row_index, col_index = self._compute_pk_solution_index(evaluated.index, evaluated.wrt)
                return EvaluatedExpr(
                    v=f"{name}({row_index}, {col_index})", typ=ValueType.VALUE_TYPE_DOUBLE, token=expr
                )
            else:
                self.__raise(expr, NotImplementedError("Subscript {0} is not supported".format(expr)))

            return EvaluatedExpr(v=f"{name}[{index}]", typ=ValueType.VALUE_TYPE_DOUBLE, token=expr)

        self.__raise(expr, NotImplementedError("Expression {0} is not supported".format(expr)))

    def _do_eval_constant(self, constant: ast.Constant) -> EvaluatedExpr:
        """处理常数"""
        v: str
        typ: ValueType = ValueType.from_val(constant.value)

        if typ in [ValueType.VALUE_TYPE_DOUBLE, ValueType.VALUE_TYPE_INT, ValueType.VALUE_TYPE_LONG]:
            v = str(constant.value)
        elif typ == ValueType.VALUE_TYPE_STRING:
            v = constant.value
        else:
            self.__raise(constant, NotImplementedError("Constant {0} is not supported".format(constant)))

        return EvaluatedExpr(v=v, typ=typ, token=constant)

    def _do_eval_op(self, operator: ast.operator) -> EvaluatedExpr:
        """处理二元表达式的符号 +, -, *, /"""
        if isinstance(operator, ast.Add):
            return EvaluatedExpr(v="+", token=operator)
        elif isinstance(operator, ast.Sub):
            return EvaluatedExpr(v="-", token=operator)
        elif isinstance(operator, ast.Mult):
            return EvaluatedExpr(v="*", token=operator)
        elif isinstance(operator, ast.Div):
            return EvaluatedExpr(v="/", token=operator)
        elif isinstance(operator, ast.Pow):
            self.__raise(operator, ValueError("_do_translate_op cannot handle power"))

        self.__raise(operator, NotImplementedError("operator {0} is not supported".format(operator)))

    def _do_eval_boolop(self, bool_operator: ast.boolop) -> EvaluatedExpr:
        if isinstance(bool_operator, ast.And):
            return EvaluatedExpr(v="&&", token=bool_operator)
        elif isinstance(bool_operator, ast.Or):
            return EvaluatedExpr(v="||", token=bool_operator)

        self.__raise(bool_operator, NotImplementedError("尚不支持 bool_operator {0}".format(bool_operator)))

    def _do_eval_unaryop(self, unary_operator: ast.unaryop) -> EvaluatedExpr:
        """处理正负号 +, -"""
        if isinstance(unary_operator, ast.UAdd):
            return EvaluatedExpr(v="+", token=unary_operator)
        elif isinstance(unary_operator, ast.USub):
            return EvaluatedExpr(v="-", token=unary_operator)

        self.__raise(unary_operator, NotImplementedError("尚不支持 unary_operator {0}".format(unary_operator)))

    def _do_eval_cmpop(self, compare_operator: ast.cmpop) -> EvaluatedExpr:
        """处理比较的符号 ==, >, >=, <, <=, !="""
        if isinstance(compare_operator, ast.Eq):
            return EvaluatedExpr(v="==", token=compare_operator)
        elif isinstance(compare_operator, ast.Gt):
            return EvaluatedExpr(v=">", token=compare_operator)
        elif isinstance(compare_operator, ast.GtE):
            return EvaluatedExpr(v=">=", token=compare_operator)
        elif isinstance(compare_operator, ast.Lt):
            return EvaluatedExpr(v="<", token=compare_operator)
        elif isinstance(compare_operator, ast.LtE):
            return EvaluatedExpr(v="<=", token=compare_operator)
        elif isinstance(compare_operator, ast.NotEq):
            return EvaluatedExpr(v="!=", token=compare_operator)
        self.__raise(compare_operator, NotImplementedError("尚不支持 compare_operator {0}".format(compare_operator)))

    def _do_translate_func_signature(self, func_def: ast.FunctionDef, signature: FuncSignature) -> list[str]:
        """翻译函数签名"""
        translated: list[str] = []
        if self.__locatable(func_def):
            translated.append(f"// {self._source_code_lines[func_def.lineno - 1].strip()}")

        func_signature_args: list[str] = []
        for arg_name, arg_type in signature.args.items():
            signature_arg = f"{arg_type.to_cc_type()} {arg_name}"
            func_signature_args.append(signature_arg)

        func_signature_args_str = ", ".join(func_signature_args)

        translated.append(f"Result<{signature.return_type.to_cc_type()}> {func_def.name}({func_signature_args_str})")

        return translated

    def _extract_func_signature(self, func_def: ast.FunctionDef) -> FuncSignature:
        self.__raise(func_def, NotImplementedError("暂不支持用户自定义 function"))

    def _inject_self_to_signature(self, signature: FuncSignature) -> FuncSignature:
        """向函数签名中添加 self context"""
        signature_ = deepcopy(signature)
        # if "self" not in signature.args.keys():
        #     # "self" is a special case to handle
        #     signature_.args = {
        #         "self": ValueType.VALUE_TYPE_VEC_REF, **signature_.args
        #     }

        signature_.args = {
            DOUBLE_CONTEXT_VARNAME: ValueType.VALUE_TYPE_DOUBLE_CONTEXT_REF,
            STRING_CONTEXT_VARNAME: ValueType.VALUE_TYPE_STRING_CONTEXT_REF,
        }

        if self._module_type == "ode":
            signature_.args[CMT_CONTEXT_VARNAME] = ValueType.VALUE_TYPE_CMT_SYS_REF

        signature_.args.update(**signature.args)

        return signature_

    def _inject_result_container_to_signature(self, signature: FuncSignature) -> FuncSignature:
        """向函数签名中添加 __local 和 __container 参数"""
        # if "__local" in signature.args.keys():
        #     raise ValueError("__local 是函数的预留字段，不能够用于函数签名")

        if "__container" in signature.args.keys():
            raise ValueError("__container 是函数的预留字段，不能够用于函数签名")

        # signature.args["__local"] = ValueType.VALUE_TYPE_CONTEXT_REF
        signature.args["__container"] = ValueType.VALUE_TYPE_VEC_REF
        return signature

    def _assert_func_signature_is_valid(self, func_def: ast.FunctionDef, signature: FuncSignature) -> None:
        """检查函数签名是否合法"""
        for arg in func_def.args.args:
            if arg.arg not in signature.args.keys() and arg.arg != "self":
                self.__raise(arg, ValueError("提供的函数签名 {0} 没有包含 {1}，无法处理对应参数".format(signature, arg.arg)))

        for arg in func_def.args.kwonlyargs:
            self.__raise(arg, NotImplementedError("尚未实现 kwarg 的翻译"))

    def _compute_ode_index(self, cmt: Compartment, wrt: SymVar | None = None, wrt2nd: SymVar | None = None) -> int:
        n_cmt = len(self._ordered_cmts)
        n_eta = len(self._etas)
        cmt_index = self._ordered_cmts.index(cmt)
        if wrt is None:
            return cmt_index
        elif wrt2nd is None:  # only wrt is given
            if isinstance(wrt, Eta):
                return n_cmt + cmt_index * n_eta + self._etas.index(wrt)
            else:
                raise TypeError("wrt is not valid {0}".format(wrt))
        else:
            if isinstance(wrt, Eta) and isinstance(wrt2nd, Eta):
                comb_index = self._eta_combs.index((wrt, wrt2nd))
                if comb_index == -1:
                    comb_index = self._eta_combs.index((wrt2nd, wrt))

                if comb_index == -1:
                    raise IndexError("wrt pair not found")

                return n_cmt + n_cmt * n_eta + cmt_index * len(self._eta_combs) + comb_index
            else:
                raise TypeError("wrt and wrt2nd is not valid {0}, {1}".format(wrt, wrt2nd))

    def _compute_pk_param_key(self,
                                param_name: str,
                                wrt: SymVar | None = None,
                                wrt_2nd: SymVar | None = None) -> tuple[str, tuple[int, int]]:
        if wrt is None:
            return param_name, (0, 0)
        elif wrt_2nd is None and isinstance(wrt, Eta):
            return param_name, (0, 1 + self._etas.index(wrt))
        else:
            raise TypeError("wrt is not valid {0}".format(wrt))

    def _compute_pk_solution_index(self, cmt_index: int | None = None, wrt: SymVar | None = None) -> tuple[int, int]:
        if cmt_index is None and wrt is None:
            return (0, 0)
        elif cmt_index is not None and wrt is None:  # looking for A(cmt_index)
            return (1 + cmt_index, 0)
        elif cmt_index is None and wrt is not None and isinstance(wrt, Eta):  # looking for F_wrt_ETA(i)
            return (0, 1 + self._etas.index(wrt))
        elif cmt_index is not None and wrt is not None and isinstance(wrt, Eta):  # looking for A(cmt_index)_wrt_ETA(i)
            return (1 + cmt_index, 1 + self._etas.index(wrt))
        else:
            raise TypeError("wrt is not valid {0}".format(wrt))

    def _retrieve_vars_from_self(self) -> list[str]:
        """生成代码从 self context 中获取变量 / 协变量"""
        retrieved_vars: list[str] = ["// 变量"]
        for var_name in self._var_context.keys():
            retrieved_vars.append(f"{mask_self_attr(var_name)} = {DOUBLE_CONTEXT_VARNAME}[\"{var_name}\"];")

        if self._module_type == "ode":
            retrieved_vars.extend(["", "// 房室"])
            for cmt_name in self._cmt_context.keys():
                retrieved_vars.append(f"{mask_self_attr(cmt_name)} = &{CMT_CONTEXT_VARNAME}[\"{cmt_name}\"];")

        retrieved_vars.extend(["", "// 数据集中的协变量"])
        for data_indexer_name, data_indexer_obj in self._dat_context.items():
            ctx_name: str

            if data_indexer_obj.typ == "continuous":
                ctx_name = DOUBLE_CONTEXT_VARNAME
            elif data_indexer_obj.typ == "categorical":
                ctx_name = STRING_CONTEXT_VARNAME
            else:
                raise TypeError()

            retrieved_vars.append(f"{mask_self_attr(data_indexer_name)} = {ctx_name}[\"{data_indexer_name}\"];")

        return retrieved_vars
