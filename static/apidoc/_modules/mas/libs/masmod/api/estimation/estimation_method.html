@dataclasses.dataclass
class BaseFitOptions:
    print: int = 1
    """Print details for every {print} iterations, note this only works when `print_type='details'`. Default to `1`"""

    print_type: typing.Literal["details", "progress"] = "details"
    """Print iteration with details (`x` and `f`) or progress (progress-bar like visualization). Default to `'details'`"""

    log_level: typing.Literal["no", "info", "trace"] = "info"
    """Level to print log, if `log_level='no'`, nothing will be print. Default to `'info'`"""

    maxiter: int = 2000
    """Max number of iterations. Default to `2000`"""

    xtol: float = 1e-6
    """Converging `x` tolerance. Default to `1e-6`"""

    ftol: float = 1e-6
    """Converging `f` (objective function) tolerance. Default to `1e-6`"""


@dataclasses.dataclass
class LaplacianLikeAlgorithmSettings(BaseFitOptions):
    minimize_method: typing.Literal["nm", "powell", "bobyqa", "lbfgs", "lbfgs_b"] = "bobyqa"
    _variant: typing.Literal["FO", "FOCEi", "Laplace"] = "FOCEi"

[docs]def SAEM(
    seed: int = 46,
    n_burnin: int = 5,
    n_chain: int = 2,
    n_mode1: int = 2,
    n_mode2: int = 0,
    n_mode3: int = 0,
    n_mode4: int = 0,
    kappa_min: float = 0.01,
    kappa_max: float = 100.0,
    negative_feedback: float = 0.4,
    optimal_acceptance: float = 0.3,
    n_stochastic: int = 200,
    n_accumulative: int = 100,
    sa_omega: float = 0.95,
    sa_sigma: float = 1.0,
    alpha: float = 0.7,
    is_auto_converge: bool = True
) -> SAEMEstimationSettings:
    """
    SAEM Estimation Method (Stochastic Approximation Expectation Maximum)

    Args:
        seed (int, optional): Seed for random number generator. Default to `46`
        n_burnin (int, optional): Number of iterations to do in burnin phase. Default to `5`
        n_chain (int, optional): Number of parallel Markov Chain. Default to `2`
        n_mode1 (int, optional): Number of iterations in Mode1 for MC sampling. Default to `2`
        n_mode2 (int, optional): Number of iterations in Mode2 for MC sampling. Default to `0`
        n_mode3 (int, optional): Number of iterations in Mode3 for MC sampling. Default to `2`
        n_mode4 (int, optional): Number of iterations in Mode4 for MC sampling. Default to `0`
        kappa_min (float, optional): Minimum adjust factor for MC sampling. Default to `0.01`
        kappa_max (float, optional: Maximum adjust factor for MC sampling. Default to `100.0`
        negative_feedback (float, optional): Negative feedback rate for MC sampling. Default to `0.4`
        optimal_acceptance (float, optional): Optimal rate of acceptance for MC sampling. Default to `0.3`
        n_stochastic (int, optional): Number of iterations in stochastic approximation phase. Default to `200`
        n_accumulative (int, optional): Number of iterations in accumulative approximation phase. Default to `100`
        sa_omega (float, optional): Decreasing rate Simulated-Annealing for Omega. Default to `0.95`
        sa_sigma (float, optional): Decreasing rate Simulated-Annealing for Sigma. Default to `1.0`
        alpha (float, optional): Learning rate in accumulative approximation phase. Default to `0.7`
        is_auto_converge (bool, optional): Whether to use auto converge criterial. Default to `True`
    """
    return SAEM(
        seed=seed,
        n_burnin=n_burnin,
        n_chain=n_chain,
        n_mode1=n_mode1,
        n_mode2=n_mode2,
        n_mode3=n_mode3,
        n_mode4=n_mode4,
        kappa_min=kappa_min,
        kappa_max=kappa_max,
        negative_feedback=negative_feedback,
        optimal_acceptance=optimal_acceptance,
        n_stochastic=n_stochastic,
        n_accumulative=n_accumulative,
        sa_omega=sa_omega,
        sa_sigma=sa_sigma,
        alpha=alpha,
        is_auto_converge=is_auto_converge
    )
