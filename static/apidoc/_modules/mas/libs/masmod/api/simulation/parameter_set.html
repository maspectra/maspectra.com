[docs]    def to_dict(self) -> dict[str, float | Theta | Eta | Eps]:
        namespace = self.__dict__
        for variable_name, variable_obj in namespace.items():
            if isinstance(variable_obj, SymVar):
                variable_obj.name = variable_name

        parameters: dict[str, float | Theta | Eta | Eps] = {}

        for variable_name, variable_obj in namespace.items():
            if isinstance(variable_obj, Theta):
                parameters[variable_name] = variable_obj
            elif isinstance(variable_obj, Eta):
                parameters[variable_name] = variable_obj
            elif isinstance(variable_obj, Eps):
                parameters[variable_name] = variable_obj
            else:
                raise KeyError(f"{variable_name} is not a instance of Theta、Eta or Eps")

        return parameters



def reorganize_theta(mod: InterpretedModule, parameters: dict[str,
                                                                float | Theta | Eta | Eps]) -> npt.NDArray[np.float_]:
    theta_values: typing.List[float] = []
    for theta in mod.thetas:
        name = theta.name
        if name not in parameters.keys():
            raise ValueError(f"{name} not found in parameter set")
        p_theta = parameters[name]
        if isinstance(p_theta, int | float):
            theta_values.append(p_theta)
        elif isinstance(p_theta, Theta):
            theta_values.append(p_theta.init_value)
    return np.array(theta_values)


def reorder_matrix(
    prev_param_names: typing.List[str], reordered_param_names: typing.List[str], matrix: npt.NDArray[np.float_]
) -> npt.NDArray[np.float_]:

    set_reordered_param_names = set(reordered_param_names)
    set_prev_param_names = set(prev_param_names)

    if len(matrix.shape) != 2:
        raise ValueError("`sigma` should be a matrix")

    if matrix.shape[0] != matrix.shape[1] != len(prev_param_names):
        raise ValueError("The shape of the `sigma` matrix is not consistent with the size of `epsilon_names`")

    if set_prev_param_names < set_reordered_param_names:
        raise ValueError(
            "The `epsilon` parameters defined in the `PopModel` {} are not provided".
            format(set_reordered_param_names - set_prev_param_names)
        )

    if set_prev_param_names > set_reordered_param_names:
        raise ValueError(
            "The provided `epsilon` parameters {} are not defined in the `PopModel`".
            format(set_prev_param_names - set_reordered_param_names)
        )

    # change the block matrix by row/column order
    order: list[int] = list()
    epsilon_name: str
    for epsilon_name in reordered_param_names:
        for sigma_index, name in enumerate(prev_param_names):
            if epsilon_name == name:
                order.append(sigma_index)

    indexer: tuple[npt.NDArray[np.int32], npt.NDArray[np.int32]] = np.ix_(order, order)

    return matrix[indexer]


def reorganize_omega(mod: InterpretedModule, parameters: dict[str,
                                                                float | Theta | Eta | Eps]) -> npt.NDArray[np.float_]:

    ordered_eta_names = []
    for eta in mod.etas:
        name = eta.name
        ordered_eta_names.append(name)

    combined_param_names: typing.List[str] = []
    visited_id = []
    omega_matrixs: typing.List[npt.NDArray[np.float_]] = []
    for param_name, param in parameters.items():
        if isinstance(param, Eta):
            eta_instance_id = id(param.omega)
            if eta_instance_id not in visited_id:
                visited_id.append(id(param))
                same_id_names = [param_name]

                same_id_etas: typing.List[Eta] = []
                same_etas_instance_index = []
                same_etas_instance_index.append(param._get_index())
                omega_matrixs.append(param.omega.init_values)
                # 找其他同 id 的 eta
                for find_param_name, find_param in parameters.items():
                    if isinstance(find_param, Eta):
                        if id(find_param) not in visited_id and id(find_param.omega) == eta_instance_id:
                            visited_id.append(id(find_param.omega))
                            same_etas_instance_index.append(find_param._get_index())
                            same_id_names.append(find_param_name)
                            same_id_etas.append(find_param)

                combined_param_names.extend(np.array(same_id_names)[np.array(same_etas_instance_index)].tolist())

    omega_block_matrix: npt.NDArray[np.float_] = block_diag(*omega_matrixs)
    return reorder_matrix(
        prev_param_names=combined_param_names, reordered_param_names=ordered_eta_names, matrix=omega_block_matrix
    )


def reorganize_sigma(mod: InterpretedModule, parameters: dict[str,
                                                                float | Theta | Eta | Eps]) -> npt.NDArray[np.float_]:

    ordered_epsilon_names = []
    for epsilon in mod.epsilons:
        name = epsilon.name
        ordered_epsilon_names.append(name)

    combined_param_names: typing.List[str] = []
    visited_id = []
    sigma_matrixs: typing.List[npt.NDArray[np.float_]] = []
    for param_name, param in parameters.items():

            if epsilon_instance_id not in visited_id:
                visited_id.append(id(param))
                same_id_names = [param_name]

                same_id_epsilons: typing.List[Eps] = []
                same_epsilons_instance_index = []
                same_epsilons_instance_index.append(param._get_index())
                sigma_matrixs.append(param.sigma.init_values)
                # 找其他同 id 的 epsilon
                for find_param_name, find_param in parameters.items():
                    if isinstance(find_param, Eps):
                        if id(find_param) not in visited_id and id(find_param.sigma) == epsilon_instance_id:
                            visited_id.append(id(find_param.sigma))
                            same_epsilons_instance_index.append(find_param._get_index())

                            same_id_epsilons.append(find_param)

    sigma_block_matrix: npt.NDArray[np.float_] = block_diag(*sigma_matrixs)
    return reorder_matrix(
        prev_param_names=combined_param_names, reordered_param_names=ordered_epsilon_names, matrix=sigma_block_matrix
    )
