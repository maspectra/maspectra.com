def simulate(
    mod: InterpretedModule,
    model_ptr: DittoPopModel,
    data: ModelData,
    parameters: typing.Dict[str, float | Theta | Eta | Eps] | ParameterSet | None = None,
    seed: int = 1234
) -> SimulationResult:
    time_stamp = datetime.datetime.now()
    if isinstance(data, ModelData):
        simulate_data = data
    else:
        raise ValueError("simulate data argument must be ModelData")

    if "mdv" in simulate_data.kwd_mapping:
        mdv_column = simulate_data.kwd_mapping["mdv"]
        concate_data = simulate_data.sampling.reset_index(drop=True)
        concate_data = concate_data[concate_data[mdv_column] == 0].reset_index(drop=True)
    else:
        concate_data = simulate_data.sampling.reset_index(drop=True)

    subject = concate_data.reserved_colnames.get("subject")
    idv = concate_data.reserved_colnames.get("idv")
    if not subject:
        raise ValueError("Data required subject mapping")
    if not idv:
        raise ValueError("Data required regressor mapping")

    subjects = concate_data[subject].unique()

    if isinstance(parameters, ParameterSet):
        parameters = parameters.to_dict()

    if parameters:
        # 取 theta
        theta_values = reorganize_theta(mod, parameters)
        # 取 eta
        eta_values = reorganize_omega(mod, parameters)
        # 取 eps
        eps_values = reorganize_sigma(mod, parameters)

    else:
        pop_params = PopParams.from_ditto(mod=mod, ditto_pop_params=model_ptr.init_pop_params())
        # 取 theta
        theta_values = pop_params.theta
        # 取 eta
        eta_values = pop_params.omega
        # 取 eps
        eps_values = pop_params.sigma

    rng = np.random.default_rng(seed=seed)
    etas_by_id = [
        rng.multivariate_normal(mean=np.zeros(eta_values.shape[0]), cov=eta_values) for _ in range(len(subjects))
    ]

    eps_by_obs_by_id = [
        rng.multivariate_normal(
            mean=np.zeros(eps_values.shape[0]),
            cov=eps_values,
            size=len(concate_data[concate_data[subject] == index_subject])
        ) for index_subject in subjects
    ]

    # if mod.typ == "pred" and len(simulate_data.dosing) > 0:
    #     warnings.warn("`event` is not defined in the `PopModel`，the argument is ignored in simulation", UserWarning)
    #     simulate_data = simulate_data.sampling

    obs_col_names = check_model_data(simulate_data, mod)

    data_by_id = _to_ditto_individual_data(simulate_data, mod=mod, covariate_names=obs_col_names)

    df_simulation = typing.cast(
        pd.DataFrame,
        model_ptr.simulate(
            theta=theta_values, eta_by_id=etas_by_id, eps_by_id_by_obs=eps_by_obs_by_id, data_by_id=data_by_id
        )
    )
    # FIXME: 当 simulated_data 为 ModelData 时，合并会校验大小写问题
    simulated_data_ = typing.cast(MasDataFrame, pd.concat([MasDataFrame(concate_data), df_simulation], axis=1))
    simulated_data_.units = simulate_data.units
    simulated_data_.kwd_mapping = simulate_data.kwd_mapping

    elapsed_time = datetime.datetime.now() - time_stamp

    return SimulationResult(
        model_ptr=model_ptr,
        mod=mod,
        elapsed_time=elapsed_time,
        simulated_df=simulated_data_,
        # TODO @zhenghan: 这里对 ID、idv、dv 的设置是 hardwired
        simulation_columns=[subject, idv] + df_simulation.columns.to_list(),
        seed=seed
    )

    def __init__(
        self,
        model_ptr: DittoPopModel,
        mod: InterpretedModule,
        elapsed_time: datetime.timedelta,
        simulated_df: MasDataFrame,
        simulation_columns: list[str],
        seed: int
    ) -> None:
        # TODO @zhenghan: 需要传入 Scenario 对象，其中包括了 obs, events, 和 mod 信息
        self.__model_ptr = model_ptr
        self.__mod = mod
        self.__elapsed_time = elapsed_time
        self.__timestamp = datetime.datetime.now()
        self.__simulation_columns = simulation_columns

        # 模拟结果
        self.__simulated_df: MasDataFrame = simulated_df

        self.__seed = seed

    @property
    def y(self) -> pd.Series[float]:
        series_y: pd.Series[float] = self.__simulated_df['_Y_']
        return series_y

    def __repr__(self) -> str:
        return "Simulation - {}".format(self.__timestamp)

    def _simulate_meta_as_tabular_data(self) -> tuple[list[str], list[tuple[str | int | float, ...]]]:

        # samples
        kwd_mapping = self.__simulated_df.kwd_mapping
        n_subject = self.__simulated_df[kwd_mapping["id"]].nunique()
        n_obs = self.__simulated_df.shape[0]

        fmt_elapsed_time = format_timedelta(self.__elapsed_time)
        return ["N Subject", "N Obs", "SEED", "CPU Elapsed Time"], [(n_subject, n_obs, self.__seed, fmt_elapsed_time)]

    def __str__(self) -> str:
        # columns = [col for col in self.simulated_df if col not in self.__simulation_columns]
        columns = self.__simulation_columns

        simulate_meta_headers, simulate_meta_data = self._simulate_meta_as_tabular_data()
        meta_summary = tabulate(
            tabular_data=simulate_meta_data,
            headers=simulate_meta_headers,
            tablefmt="pipe",
            numalign="right",
            stralign="left",
            colalign=["left", "left", "left"]
        )

        simulate_summary = tabulate(
            tabular_data={
                "Variable": [col if col != "_Y_" else "Y" for col in columns],
                "Min": [self.simulated_df[col].min() for col in columns],
                "Max": [self.simulated_df[col].max() for col in columns],
                "Mean": [self.simulated_df[col].mean() for col in columns],
                "SD": [self.simulated_df[col].std() for col in columns],
            },
            headers="keys",
            tablefmt="pipe",
            numalign="right",
            stralign="left",
            colalign=["left", "left", "left", "left"]
        )
        out = f"""================= Summary =================
*** Simulate Summary ***
{meta_summary}

*** Parameters Summary ***
{simulate_summary}

"""
        return out

    @property
    def simulated_df(self) -> pd.DataFrame:
        """Simulate dataframe result"""
        return self.__simulated_df

    def _ipython_display_(self) -> None:
        from IPython.display import display_markdown

        columns = self.__simulation_columns

        simulate_meta_headers, simulate_meta_data = self._simulate_meta_as_tabular_data()\

        display_markdown("#### Simulate Summary", raw=True)
        meta_summary = tabulate(
            tabular_data=simulate_meta_data,
            headers=simulate_meta_headers,
            tablefmt="pipe",
            numalign="right",
            stralign="left",
            colalign=["left", "left", "left"]
        )
        display_markdown(meta_summary, raw=True)

        display_markdown("\n#### Parameter Summary", raw=True)
        simulate_summary = tabulate(
            tabular_data={
                "Variable": [col if col != "__Y__" else "Y" for col in columns],
                "Min": [self.simulated_df[col].min() for col in columns],
                "Max": [self.simulated_df[col].max() for col in columns],
                "Mean": [self.simulated_df[col].mean() for col in columns],
                "SD": [self.simulated_df[col].std() for col in columns],
            },
            headers="keys",
            tablefmt="pipe",
            numalign="right",
            stralign="left",
            colalign=["left", "left", "left", "left"]
        )
        display_markdown(simulate_summary, raw=True)


[docs]    def plot_individual(
        self,
        y_axis: str = "_Y_",
        x_scale: Literal["linear", "log"] = "linear",
        y_scale: Literal["linear", "log"] = "linear"
    ) -> SimulateIndividualPlot:
        """individual plot

        Args:
            y_axis (str, optional): the column of y axis data. Defaults to "__Y__".
            x_scale (Literal["linear", "log"], optional): x scale type. Defaults to "linear".
            y_scale (Literal["linear", "log"], optional): y scale type. Defaults to "linear".

        Raises:
            ValueError:

        Returns:
            SimulateIndividualPlot: Simulate individual plot object
        """
        plot_data = self.__simulated_df.copy(deep=True)
        kwd_mapping = self.__simulated_df.kwd_mapping
        if kwd_mapping.get("time"):
            idv_name = kwd_mapping["time"]
        elif kwd_mapping.get("idv"):
            idv_name = kwd_mapping["idv"]
        else:
            raise ValueError("simulate result data has no idv mapping")
        subject = kwd_mapping.get("id")
        return SimulateIndividualPlot(
            plot_data, subject_id_name=subject, idv_name=idv_name, dv_name=y_axis, x_scale=x_scale, y_scale=y_scale
        )
