def _retrieve_thetas(mod: InterpretedModule) -> DittoThetas:
    names: list[str] = []
    init_values = np.array([], dtype=np.float_)
    lower_bounds = np.array([], dtype=np.float_)
    upper_bounds = np.array([], dtype=np.float_)
    fixed = np.array([], dtype=int)

    for _theta in mod.thetas:
        names.append(_theta.name)
        init_values = np.append(init_values, _theta.init_value)
        _lower, _upper = _theta.bounds
        lower_bounds = np.append(lower_bounds, _lower)
        upper_bounds = np.append(upper_bounds, _upper)
        fixed = np.append(fixed, 1 if _theta.fixed else 0)

    return DittoThetas(
        names=names, init_values=init_values, lower_bounds=lower_bounds, upper_bounds=upper_bounds, fixed=fixed
    )


def _retrieve_omega(mod: InterpretedModule) -> DittoOmega:
    visited_ids: set[int] = set()
    names: list[str] = []
    init_values: npt.NDArray[np.float_] = np.empty([0, 0], dtype=np.float_)
    fixed = np.empty([0, 0], dtype=np.int_)

    for _eta in mod.etas:
        names.append(_eta.name)

        if id(_eta.omega) not in visited_ids:
            visited_ids.add(id(_eta.omega))
            init_values = block_diagonal(left=init_values, right=_eta.omega.init_values, fill=np.nan, dtype=np.float_)
            fixed = block_diagonal(
                left=fixed,
                right=_eta.omega.fixed * np.ones(_eta.omega.init_values.shape, dtype=np.int_),
                fill=0,
                dtype=np.int_
            )

    return DittoOmega(names=names, init_values=init_values, fixed=fixed)


def _retrieve_sigma(mod: InterpretedModule) -> DittoSigma:
    visited_ids: set[int] = set()
    names: list[str] = []
    init_values: npt.NDArray[np.float_] = np.empty([0, 0], dtype=np.float_)
    fixed = np.empty([0, 0], dtype=np.int_)

    for _eps in mod.epsilons:
        names.append(_eps.name)

        if id(_eps.sigma) not in visited_ids:
            visited_ids.add(id(_eps.sigma))
            init_values = block_diagonal(left=init_values, right=_eps.sigma.init_values, fill=np.nan, dtype=np.float_)
            fixed = block_diagonal(
                left=fixed,
                right=_eps.sigma.fixed * np.ones(_eps.sigma.init_values.shape, dtype=np.int_),
                fill=0,
                dtype=np.int_
            )

    return DittoSigma(names=names, init_values=init_values, fixed=fixed)


def _retrieve_covariate(mod: InterpretedModule, data: ModelData | None = None) -> dict[str, str]:
    # TODO @zhenghan: 这样处理似乎不好，会导致模拟比建模少一层 mapping
    if data is None:
        return {indexer_name: mod.column_context[indexer_name].col_name for indexer_name in mod.column_context.keys()}
    else:
        data_indexer_mp: dict[str, str] = {}
        for indexer_name, indexer_obj in mod.column_context.items():
            if indexer_obj.col_name not in data:
                raise KeyError("data 中没有 DataIndexer 指定的列名 {}".format(indexer_obj.col_name))
            data_indexer_mp[indexer_name] = indexer_obj.col_name

        return data_indexer_mp


def _retrieve_cmt_system(mod: InterpretedModule) -> DittoCompartmentSystem:
    cmts: list[DittoCompartment] = []
    for name, cmt in mod.cmt_context.items():
        A0 = 0.
        if isinstance(cmt.init_value.expr, int | float):
            A0 = cmt.init_value.expr
        cmts.append(
            DittoCompartment(label=name, init_value=A0, default_dose=cmt.default_dose, default_obs=cmt.default_obs)
        )

    return DittoCompartmentSystem(compartments=cmts)


def _do_estimation(
    mod: InterpretedModule,
    estimator: DittoEstimator,
    ptr: DittoPopModel,
    estimation: AnyEstimationSettings,
    pop_params: PopParams,
) -> EstimationResult:
    if isinstance(estimation, typing.Callable):
        estimation = estimation()
    est_result: DittoEstimationResult
    time_stamp = datetime.datetime.now()
    if isinstance(estimation, LaplacianLikeAlgorithmSettings):
        # Laplace Like
        ditto_minimize_method: PyModelFitMethods
        if estimation.minimize_method == "nm":
            ditto_minimize_method = PyModelFitMethods.PY_MODEL_FIT_METHOD_NELDER_MEAD
        elif estimation.minimize_method == "powell":
            ditto_minimize_method = PyModelFitMethods.PY_MODEL_FIT_METHOD_POWELL
        elif estimation.minimize_method == "bobyqa":
            ditto_minimize_method = PyModelFitMethods.PY_MODEL_FIT_METHOD_BOBYQA
        elif estimation.minimize_method == "lbfgs" or estimation.minimize_method == "lbfgs_b":
            ditto_minimize_method = PyModelFitMethods.PY_MODEL_FIT_METHOD_LBFGS_B
        else:
            raise NotImplementedError("Invalid `minimize_method`")

        settings_dict: dict[str, int | float] = {
            "ftol": estimation.ftol, "xtol": estimation.xtol, "print": estimation.print, "maxiter": estimation.maxiter
        }

        log_level: int
        if estimation.log_level == "no":
            log_level = 0
        elif estimation.log_level == "info":
            log_level = 1
        elif estimation.log_level == "trace":
            log_level = 2
        else:
            raise ValueError("Invalid `log_level`")
        settings_dict["log_level"] = log_level

        print_type: int
        if estimation.print_type == "details":
            print_type = 0
        elif estimation.print_type == "progress":
            print_type = 1
        else:
            raise ValueError("Invalid `print_type`")
        settings_dict["print_type"] = print_type

        if estimation._variant == "FO":
            if mod.pred_variant != "y":
                raise ValueError("Using First Order Estimation Method should give a Y Predictor")
            est_result = estimator.first_order_estimate(
                pop_params=pop_params, minimize_method=ditto_minimize_method, settings_dict=settings_dict
            )
        elif estimation._variant == "FOCEi":
            if mod.pred_variant != "y":
                raise ValueError("Using First Order Conditional Estimation Method should give a Y Predictor")
            est_result = estimator.first_order_conditional_estimation(
                pop_params=pop_params, minimize_method=ditto_minimize_method, settings_dict=settings_dict
            )
        elif estimation._variant == "Laplace":
            if mod.pred_variant == "y":
                raise ValueError("Using Laplace Estimation Method should give a Likelihood Predictor")
            est_result = estimator.first_order_conditional_estimation(
                pop_params=pop_params, minimize_method=ditto_minimize_method, settings_dict=settings_dict
            )
        else:
            raise NotImplementedError()
    elif isinstance(estimation, SAEMEstimationSettings):
        # SAEM
        saem_settings = saem.DittoSAEMSettings(
            n_mode1=estimation.n_mode1,
            n_mode2=estimation.n_mode2,
            n_mode3=estimation.n_mode3,
            n_mode4=estimation.n_mode4,
            kappa_min=estimation.kappa_min,
            kappa_max=estimation.kappa_max,
            negative_feedback=estimation.negative_feedback,
            optimal_acceptance=estimation.optimal_acceptance,
            n_stochastic=estimation.n_stochastic,
            n_accumulative=estimation.n_accumulative,
            sa_omega=estimation.sa_omega,
            sa_sigma=estimation.sa_sigma,
            alpha=estimation.alpha,
            is_auto_converge=estimation.is_auto_converge
        )
        est_result = estimator.stochastic_approx_em(
            pop_params=pop_params,
            saem_settings=saem_settings,
            n_burnin=estimation.n_burnin,
            n_chain=estimation.n_chain,
            seed=estimation.seed
        )
    else:
        raise NotImplementedError("Invalid Estimation")
    elapsed_time = datetime.datetime.now() - time_stamp

    return EstimationResult(
        model_ptr=ptr,
        mod=mod,
        ditto_est_results=est_result,
        elapsed_time=elapsed_time,
    )


def _fit(
    data: ModelData | None,
    init_pop_params: PopParams,
    mod: InterpretedModule,
    estimator: DittoEstimator,
    ptr: DittoPopModel,
    estimation: AnyEstimationSettings | typing.Sequence[AnyEstimationSettings],
    pop_params: PopParams | ParameterSet | None = None,
) -> FitResult:
    """Model Fitting Entry Point

    Args:
        estimation (AnyEstimationSettings | typing.Iterable[AnyEstimationSettings]): Estimation Controls. It is possible to go with single estimation or a set of estimations.
        pop_params (PopParams | None, optional): Estimating Initial Population Parameters. Default to the values defined in mod.

    Returns:
        FitResult: Collections for Estimation Results

    Examples:
        >>> single_estimation_result = pop_model.fit(FOCEi(print_type="progress"))
        >>> multiple_estimation_result = pop_model.fit([
                FOCEi(print=100, log_level='info'),
                SAEM(is_auto_converge=True)
            ])
    """
    if data is None:
        raise RuntimeError("The model cannot be fitted without specifying data")

    parameters: PopParams
    if pop_params is None:
        parameters = init_pop_params
    else:
        if isinstance(pop_params, ParameterSet):
            params_d = pop_params.to_dict()
            theta_values = reorganize_theta(mod, params_d)
            omega_values = reorganize_omega(mod, params_d)
            sigma_values = reorganize_sigma(mod, params_d)

            parameters = PopParams(theta=theta_values, omega=omega_values, sigma=sigma_values, mod=mod)
        else:
            parameters = pop_params

    _estimations_to_do: list[AnyEstimationSettings] = []
    if isinstance(estimation, typing.Iterable):
        if len(estimation) == 0:
            raise ValueError("You must provide at least one estimation to fit model")
        _estimations_to_do = [*estimation]
    else:
        _estimations_to_do = [estimation]

    est_results: list[EstimationResult] = []
    _previous_est_result: EstimationResult | None = None
    for est_ in _estimations_to_do:
        result = _do_estimation(mod=mod, estimator=estimator, ptr=ptr, estimation=est_, pop_params=parameters)
        parameters = result.final_pop_params
        result._previous_est_result = _previous_est_result
        _previous_est_result = result
        est_results.append(result)

    return FitResult(
        model_ptr=ptr,
        data=data,
        mod=mod,
        estimator=estimator,
        est_results=tuple(est_results),
        init_pop_params=init_pop_params,
        fit_args=dict(estimation=estimation, pop_params=pop_params)
    )

    def __init__(
        self,
        mod: AnyModuleType | typing.Callable[[], AnyModuleType],
        data: pd.DataFrame | ModelData | None = None,
    ) -> None:
        if isinstance(mod, typing.Callable):
            mod = mod()
        if not isinstance(mod, Module) and \
            not isinstance(mod, InterpretedModule | RefinedModule | TranslatedModule):
            raise TypeError("'mod' is not Module Type")

        if isinstance(data, ModelData):
            pass
        elif isinstance(data, pd.DataFrame):
            data = ModelData(data)
        elif data is not None:
            raise TypeError("'data' is not Data Type")

        _logging_logger = logging.Logger("MasmodLogger")
        if not _logging_logger.hasHandlers():
            hdl = logging.StreamHandler(stream=sys.stdout)
            hdl.terminator = ""
            _logging_logger.addHandler(hdl)
            _logging_logger.setLevel(logging.INFO)
        _logger = LoggingLogger(_logging_logger)

        self._logger = _logger  # 必须保留这个指针，不然 logger 会在 __init__ 后被回收，无法正确 adapt C++ logger
        self._adaptive_logger = AdaptiveLogger(
            debug_functor=_logger.debug,
            info_functor=_logger.info,
            warn_functor=_logger.warning,
            error_functor=_logger.error,
            fatal_functor=_logger.critical
        )

        _mod: Module | InterpretedModule | RefinedModule | TranslatedModule = mod
        if not isinstance(_mod, InterpretedModule):
            _mod = interpret(_mod)

        # 校验数据
        if data is not None:
            covariate_names = check_model_data(data, _mod)
        else:
            covariate_names = []

        if not isinstance(_mod, RefinedModule):
            _mod = refine(_mod)

        if not isinstance(_mod, TranslatedModule):
            _mod = translate(_mod)

        self.__mod = _mod
        root_dir = pathlib.Path(__file__).parent.parent.parent.parent.parent
        eigen_dir = root_dir.joinpath("mas/vendor/eigen")
        compiler = CCCompiler(
            logger=_logger,
            cc_source_code=self.__mod.cc_source_code,
            src_include_dir=root_dir,
            eigen_include_dir=eigen_dir,
            temp_dir=None,
        )
        dylib_path = compiler.compile().target_file

        self.cmt_system = _retrieve_cmt_system(mod=self.__mod)

        ditto_thetas = _retrieve_thetas(mod=self.__mod)
        ditto_omega = _retrieve_omega(mod=self.__mod)
        ditto_sigma = _retrieve_sigma(mod=self.__mod)
        ditto_covariates = _retrieve_covariate(mod=self.__mod)

        if data is None:
            self.__data = None
            self._ptr = DittoPopModel(
                dylib_path=dylib_path.as_posix(),
                module_factory_func_name=CCTranslator.MODULE_FACTORY_FUNC_NAME,
                thetas=ditto_thetas,
                omega=ditto_omega,
                sigma=ditto_sigma,
                covariate_mp=ditto_covariates,
                cmt_system=self.cmt_system
            )
        else:
            self.__data = data
            _data_by_id = _to_ditto_individual_data(data, mod=self.__mod, covariate_names=covariate_names)
            # data_by_id = _to_ditto_individual_data(data, covariate_names=obs_col_names)
            self._ptr = DittoPopModel(
                dylib_path=dylib_path.as_posix(),
                module_factory_func_name=CCTranslator.MODULE_FACTORY_FUNC_NAME,
                thetas=ditto_thetas,
                omega=ditto_omega,
                sigma=ditto_sigma,
                covariate_mp=ditto_covariates,
                cmt_system=self.cmt_system,
                data_by_id=_data_by_id
            )
        self._estimator = DittoEstimator(model=self._ptr, logger=self._adaptive_logger)

    def __deepcopy__(self, memo: typing.Any) -> PopModel:
        if self in memo:

        memo[self] = dup
        return dup

[docs]    def simulate(
        self,
        data: ModelData | None = None,
        parameters: typing.Dict[str, float | Theta | Eta | Eps] | ParameterSet | None = None
    ) -> SimulationResult:

        if isinstance(data, ModelData):
            simulate_data = data
        elif data is None:
            if self.__data is None:
                raise ValueError("argument `data` cannot be None when model does not have data")
            simulate_data = self.__data

            raise ValueError("simulate data argument must be ModelData or None")
