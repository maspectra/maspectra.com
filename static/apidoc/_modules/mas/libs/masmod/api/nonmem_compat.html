[docs]def split_nonmem_data(
    df: pd.DataFrame,
    id_col: str = "ID",
    idv_col: str = "TIME",
    dv_col: str = "DV",
    amt_col: str = "AMT",
    cmt_col: str = "CMT",
    evid_col: str = "EVID",
    ii_col: str = "II",
    addl_col: str = "ADDL",
    rate_col: str = "RATE",
    dur_col: str = "DUR",
) -> tuple[ModelData, ModelData]:
    df.replace(".", value=pd.NA, inplace=True)
    df = df.infer_objects()
    if "EVID" in df.columns and "MDV" in df.columns:
        df["EVID"] = df["EVID"].astype("int")
        df["MDV"] = df["MDV"].astype("int")
        obs_df = df[(df["EVID"] == 0) & (df["MDV"] == 0)]
        event_df = df[(df["EVID"] > 0) & (df["MDV"] == 1)]
    elif "EVID" in df.columns and "MDV" not in df.columns:
        df["EVID"] = df["EVID"].astype("int")
        obs_df = df[(df["EVID"] == 0) & (~df[dv_col].isnull())]
        event_df = df[(df["EVID"] > 0) & (~df[amt_col].isnull())]
    elif "EVID" not in df.columns and "MDV" in df.columns:
        df["MDV"] = df["MDV"].astype("int")
        event_df = df[(df["MDV"] == 1) & (~df[amt_col].isnull())]
        obs_df = df[(df["MDV"] == 0) & (~df[dv_col].isnull())]
    else:
        event_df = df[(df[dv_col].isnull()) & (~df[amt_col].isnull())]
        obs_df = df[(~df[dv_col].isnull())]

    event_df.reset_index(drop=True, inplace=True)
    obs_df.reset_index(drop=True, inplace=True)

    obs_table = ModelData(
        obs_df,
        id=id_col,
        time=idv_col,
        amt=amt_col,
        cmt=cmt_col,
        evid=evid_col,
        ii=ii_col,
        addl=addl_col,
        rate=rate_col,
        dur=dur_col
    )
    dose_table = ModelData(
        event_df,
        id=id_col,
        time=idv_col,
        amt=amt_col,
        cmt=cmt_col,
        evid=evid_col,
        ii=ii_col,
        addl=addl_col,
        rate=rate_col,
        dur=dur_col
    )

    return obs_table, dose_table




[docs]def read_nonmem_data(
    path: str, ignore: str = "#", encoding: str = "utf-8", columns: List[str] | None = None
) -> pd.DataFrame:
    """read nonmem data

    Args:
        path (str): data path
        ignore (str): ignore lines that start with specific characters. default '#'.
        encoding (str): specified file encoding, default 'utf-8'
        columns (str): Specified data columns.

    Returns:
        pd.DataFrame: read nonmem data as a dataframe
    """
    # 第一遍剔除 ignore
    with open(path, encoding=encoding) as f:
        lines = f.readlines()
    file_buffer = StringIO()
    if ignore != "@":
        for line in lines:
            if not line.startswith(ignore):
                file_buffer.write(line)
    else:
        for line in lines:
            if len(line) > 0 and (line[0].isdigit()):
                file_buffer.write(line)
    data = pd.read_csv(StringIO(file_buffer.getvalue()), na_values=["."], header=None)
    unnamed_columns = [f"Unnamed{colname}" for colname in data.columns]
    n_cols = len(unnamed_columns)
    if columns:
        if len(columns) >= data.shape[0]:
            data.columns = pd.Index(columns[:n_cols])
        else:
            data.columns = pd.Index(columns + unnamed_columns[len(columns):n_cols])
    else:
        data.columns = pd.Index(unnamed_columns)
    return data



class TranslateResult(IIPythonDisplayable):

    def __init__(self, ctl_source: str, mod_source: str, work_dir: pathlib.Path | None = None) -> None:
        super().__init__()
        self.__ctl_source = ctl_source
        self.__mod_source = mod_source
        self.__work_dir = work_dir

    @property
    def ctl_source(self) -> str:
        return self.__ctl_source

    @property
    def mod_source(self) -> str:
        return self.__mod_source

    @property
    def work_dir(self) -> pathlib.Path | None:
        return self.__work_dir

    def _ipython_display_(self) -> None:
        from IPython.display import display
        display({
            "text/masmod": self.__mod_source, "text/plain": self.__mod_source
        }, raw=True)


def translate_nonmem(
    ctl_source: str | None = None,
    ctl_path: pathlib.Path | str | None = None,
    work_dir: str | pathlib.Path | None = None
) -> TranslateResult:
    """nonmem control file translate into mas model

    Args:
        ctl_source_or_path (str | pathlib.Path): nonmem control file path or source
        work_dir (str | pathlib.Path | None, optional): work directory. Defaults to None.

    Returns:
        TranslateResult: TranslateResult
    """

    if not ctl_path and not ctl_source:
        raise ValueError("both ctl source and path is not specified")
    if isinstance(work_dir, str):
        work_dir = pathlib.Path(work_dir)
    source = CTLSource(ctl_path=ctl_path, source=ctl_source)
    lex = CTLLexAnalysis(source)
    lex.analysis()
    syn = SynParser(lex)
    syn.parse()
    translator = CtlAstTranslator(syn, work_dir=work_dir)
    translator.translate()
    mas_model_str = translator.write()
    return TranslateResult(ctl_source=source.ctl_code, mod_source=mas_model_str, work_dir=work_dir)
