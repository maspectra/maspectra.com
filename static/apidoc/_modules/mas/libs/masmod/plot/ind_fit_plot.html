import math

import pandas as pd
import plotly.graph_objs as go
from plotly.subplots import make_subplots

from mas.libs.masmod.plot.abstract_groupby import AbstractGroupbyablePlotlyPlot



[docs]class IndFitPlot(AbstractGroupbyablePlotlyPlot):

    def __init__(
        self,
        data: pd.DataFrame,
        title: str,
        subject_id_name: str | None,
        meta: dict[str, str],
        idv_columns: str,
        dv_columns: str,
        pred_columns: str = "PRED",
        ipred_columns: str = "IPRED",
        label_dv: str = "DV",
        label_pred: str = "PRED",
        label_ipred: str = "IPRED",
        sub_rows: int | None = None,
        sub_cols: int | None = None,
        is_plot_subgraph: bool = True
    ) -> None:
        super().__init__(data, title, subject_id_name, meta, is_plot_subgraph)
        if (isinstance(sub_rows, int | None)):
            self._sub_rows: int | None = sub_rows
        else:
            raise TypeError("'rows' should be type int.")
        if (isinstance(sub_cols, int | None)):
            self._sub_cols: int | None = sub_cols
        else:
            raise TypeError("'cols' should be type int.")
        if (isinstance(label_dv, str)):
            self._label_dv: str = label_dv
        else:
            raise TypeError("'label dv' should be type str.")
        if (isinstance(label_pred, str)):
            self._label_pred: str = label_pred
        else:
            raise TypeError("'label pred' should be type str.")
        if (isinstance(label_ipred, str)):
            self._label_ipred: str = label_ipred
        else:
            raise TypeError("'label ipred' should be type str.")
        if (not isinstance(idv_columns, str)):
            raise TypeError("'idv columns' should be type str.")
        if (not isinstance(dv_columns, str)):
            raise TypeError("'dv columns' should be type str.")
        if (not isinstance(pred_columns, str)):
            raise TypeError("'pred columns' should be type str.")
        if (not isinstance(ipred_columns, str)):
            raise TypeError("'ipred columns' should be type str.")

        if (idv_columns in self._data.columns):
            self._idv_columns: str = idv_columns
        else:
            raise ValueError("The idv column is not in the data.")
        if (dv_columns in self._data.columns):
            self._dv_columns: str = dv_columns
        else:
            raise ValueError("The dv column is not in the data.")
        if (pred_columns in self._data.columns):
            self._pred_columns: str = pred_columns
        else:
            raise ValueError("The pred column is not in the data.")
        if (ipred_columns in self._data.columns):
            self._ipred_columns: str = ipred_columns
        else:
            raise ValueError("The ipred column is not in the data.")

    def _make_plot(self) -> go.Figure:
        subject_fit_plot_data_list_by_id: dict[str, pd.DataFrame] = dict()
        sub_title_list: list[str] = []  # 设置子图标题
        if (self._is_subplot):  # 按 id 绘制子图
            if (self._id_name is not None) and (self._id_name in self._data.columns):  # 校验 id 列名存在且在数据集中
                sub_title_front: str = str(self._id_name)  # 子图标题前半部分
                for (id_key, id_data) in self._data.groupby(self._id_name):
                    subject_fit_plot_data_list_by_id[str(id_key)] = id_data

                n_ids: int = len(subject_fit_plot_data_list_by_id)
                if (self._sub_rows is None) and (self._sub_cols is None):  # 依据用户设定校验并计算画图的行列数
                    self._sub_cols = int(math.ceil(n_ids**0.5))
                    self._sub_rows = math.ceil(n_ids / self._sub_cols)
                elif (self._sub_rows is None) and (self._sub_cols is not None):
                    if (self._sub_cols <= 0):
                        raise ValueError("The number of columns should be greater than 0.")
                    else:
                        self._sub_rows = math.ceil(n_ids / (self._sub_cols))
                elif (self._sub_rows is not None) and (self._sub_cols is None):
                    if (self._sub_rows <= 0):
                        raise ValueError("The number of rows should be greater than 0.")
                    else:
                        self._sub_cols = math.ceil(n_ids / self._sub_rows)
                elif (self._sub_rows is not None) and (self._sub_cols is not None):
                    if (self._sub_cols <= 0):
                        raise ValueError("The number of columns should be greater than 0.")
                    if (self._sub_rows <= 0):
                        raise ValueError("The number of rows should be greater than 0.")
                else:
                    raise ValueError("Incorrect rows and cols input.")

                for id_key in subject_fit_plot_data_list_by_id.keys():
                    sub_title_list.append(sub_title_front + " = " + id_key)
            else:
                raise RuntimeError(
                    "The columns used to plot the subplot are not defined, or the columns are not in the data"
                )
        else:
            subject_fit_plot_data_list_by_id[""] = self._data

            if (self._sub_rows is None) and (self._sub_cols is None):  # 不画子图的时候强制设行列
                self._sub_cols = 1
                self._sub_rows = 1
            elif (self._sub_rows is None) and (self._sub_cols is not None):
                if (self._sub_cols <= 0):
                    raise ValueError("The number of columns should be greater than 0.")
                else:
                    self._sub_rows = 1
            elif (self._sub_rows is not None) and (self._sub_cols is None):
                if (self._sub_rows <= 0):
                    raise ValueError("The number of rows should be greater than 0.")
                else:
                    self._sub_cols = 1
            elif (self._sub_rows is not None) and (self._sub_cols is not None):
                if (self._sub_cols <= 0):
                    raise ValueError("The number of columns should be greater than 0.")
                if (self._sub_rows <= 0):
                    raise ValueError("The number of rows should be greater than 0.")
            else:
                raise ValueError("Incorrect rows and cols input.")
            sub_title_list = [""]

        fig: go.Figure
        if (isinstance(self._sub_rows, int)) and (isinstance(self._sub_cols, int)):
            fig = make_subplots(
                rows=self._sub_rows,
                cols=self._sub_cols,
                subplot_titles=sub_title_list,
                x_title=self._idv_columns,
                y_title="Observation/Prediction",
                shared_xaxes=True,
                shared_yaxes=True
            )
        else:
            raise RuntimeError("'rows' and 'cols' should be type int.")

        subplot_index: int = 0
        for id_key in subject_fit_plot_data_list_by_id.keys():  # 遍历 id 画图
            id_plot_data: pd.DataFrame = subject_fit_plot_data_list_by_id[id_key]
            trace_dv = go.Scatter(
                x=id_plot_data[self._idv_columns],
                y=id_plot_data[self._dv_columns],
                mode="markers",
                name=self._label_dv,
                legendgroup=self._label_dv,
                marker=dict(symbol="circle-open", color="#4169E1", size=8),
                showlegend=False,
            )
            trace_pred = go.Scatter(
                x=id_plot_data[self._idv_columns],
                y=id_plot_data[self._pred_columns],
                mode="lines",
                name=self._label_pred,
                legendgroup=self._label_pred,
                line=dict(shape="linear", dash="solid", color="#000000"),
                showlegend=False
            )
            trace_ipred = go.Scatter(
                x=id_plot_data[self._idv_columns],
                y=id_plot_data[self._ipred_columns],
                mode="lines",
                name=self._label_ipred,
                legendgroup=self._label_ipred,
                line=dict(shape="linear", dash="dash", color="#A52A2A"),
                showlegend=False
            )

            if (subplot_index == 0):
                trace_dv["showlegend"] = True
                trace_pred["showlegend"] = True
                trace_ipred["showlegend"] = True

            row_index: int = subplot_index // self._sub_cols + 1
            col_index: int = subplot_index % self._sub_cols + 1
            fig.add_trace(trace_dv, row=row_index, col=col_index)
            fig.add_trace(trace_pred, row=row_index, col=col_index)
            fig.add_trace(trace_ipred, row=row_index, col=col_index)
            subplot_index += 1

        height: float = 250 * self._sub_rows
        if (self._sub_rows < 3):  # 当子图数量过少时，调整大小
            height = 350 * self._sub_rows
        width: float = 250 * self._sub_cols
        if (self._sub_cols < 3):  # 当子图数量过少时，调整大小
            width = 350 * self._sub_cols
        fig.update_xaxes(
            showline=True,
            linecolor='#000',
            linewidth=2,
            mirror=True,
            ticks='outside',
        )
        fig.update_yaxes(
            showline=True,
            linecolor='#000',
            linewidth=2,
            mirror=True,
            ticks='outside',
        )

        fig.update_layout(
            height=height,
            width=width,
            title={
                "text": self._title, 'y': 0.98, 'x': 0.5
            },
            title_font={"size": 20},
            plot_bgcolor='#fff'
        )
        return fig

        )
