import numpy as np
import numpy.typing as npt
import pandas as pd
import plotly.graph_objs as go
from plotly.subplots import make_subplots

from mas.libs.masmod.plot.abstract import AbstractPlotlyPlot



[docs]class IterPlot(AbstractPlotlyPlot):

    def __init__(self, data: pd.DataFrame) -> None:
        self._data: pd.DataFrame = data
        self._figure: None | go.Figure = None

    def _make_plot(self) -> go.Figure:
        data_cols: int = len(self._data.columns) - 1
        n_iter: int = self._data.shape[0]
        fig: go.Figure = make_subplots(
            rows=data_cols, cols=1, subplot_titles=self._data.columns.to_numpy()[1:data_cols + 1]
        )
        for i in range(data_cols):
            col_name: str = str(self._data.columns[i + 1])
            iter_str_arr: npt.NDArray[np.str_] = self._data["iteration"].apply(lambda x: str(x)).to_numpy()
            col_name_str_arr: npt.NDArray[np.str_] = np.repeat('<br />' + col_name + " = ", n_iter)
            col_value_str_arr: npt.NDArray[np.str_] = self._data[col_name].apply(lambda x: str("%.2e" % x)).to_numpy()
            text_arr: npt.NDArray[
                np.str_] = np.repeat("iter = ", n_iter) + iter_str_arr + col_name_str_arr + col_value_str_arr
            trace = go.Scatter(
                x=self._data["iteration"],
                y=self._data[col_name],
                mode="lines",
                name=col_name,
                text=text_arr,
                hoverinfo="text"
            )
            fig.add_trace(trace, i + 1, 1)

        fig.update_xaxes(
            showline=True,
            range=[0.9, n_iter * 1.01],
            linecolor='#000',
            linewidth=2,
            mirror=True,
            ticks='outside',
        )
        fig.update_yaxes(
            showline=True,
            linecolor='#000',
            linewidth=2,
            mirror=True,
            ticks='outside',
        )

        height: float = 200 * data_cols

        fig.update_layout(
            height=height,
            title={
                "text": "Iteration history plot", 'y': 0.98, 'x': 0.5
            },
            title_font={"size": 20},
            plot_bgcolor='#fff'
        )
        return fig
